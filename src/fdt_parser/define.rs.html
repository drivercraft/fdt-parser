11/// The Device Tree Blob magic number (0xd00dfeed).
12///
13/// This value must be present at the start of any valid Device Tree Blob.
14pub const FDT_MAGIC: u32 = 0xd00dfeed;

16/// Token type for parsing FDT structure blocks.
17///
18/// Tokens are 32-bit values that identify different elements in the
19/// device tree structure block.
20#[derive(Debug, PartialEq, Eq, Clone, Copy)]
21pub(crate) enum Token {
22    /// Begin node token (0x1)
23    BeginNode,
24    /// End node token (0x2)
25    EndNode,
26    /// Property token (0x3)
27    Prop,
28    /// No-op token (0x4)
29    Nop,
30    /// End token (0x9) - marks the end of the structure block
31    End,
32    /// Any other data (not a valid token)
33    Data,
34}
35
36impl From<u32> for Token {
37    fn from(value: u32) -> Self {
38        match value {
39            0x1 => Token::BeginNode,
40            0x2 => Token::EndNode,
41            0x3 => Token::Prop,
42            0x4 => Token::Nop,
43            0x9 => Token::End,
44            _ => Token::Data,
45        }
46    }
47}

58/// A memory reservation entry in the FDT.
59///
60/// Memory reservations specify physical memory regions that must
61/// not be overwritten by the device tree or bootloader.
62#[derive(Clone, Copy)]
63pub struct MemoryRegion {
64    /// Physical address of the reserved region
65    pub address: *mut u8,
66    /// Size of the reserved region in bytes
67    pub size: usize,
68}
69
70impl Debug for MemoryRegion {
71    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
72        f.write_fmt(format_args!(
73            "MemoryRegion {{ address: {:p}, size: {:#x} }}",
74            self.address, self.size
75        ))
76    }
77}

79/// A phandle (pointer handle) for referencing nodes in the device tree.

105/// A register entry describing a memory-mapped region.
106///
107/// The `reg` property contains one or more of these entries, each
108/// describing a address range for a device's registers.
109#[derive(Clone, Copy)]
110pub struct FdtReg {
111    /// Parent bus address
112    pub address: u64,
113    /// Child bus address
114    pub child_bus_address: u64,
115    /// Size of the region (None if not specified)
116    pub size: Option<usize>,
117}
118
119impl Debug for FdtReg {
120    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
121        f.write_fmt(format_args!("<{:#x}", self.address))?;
122        if self.child_bus_address != self.address {
123            f.write_fmt(format_args!("({:#x})", self.child_bus_address))?;
124        }
125        f.write_fmt(format_args!(", "))?;
126        if let Some(s) = self.size {
127            f.write_fmt(format_args!("{:#x}>", s))
128        } else {
129            f.write_str("None>")
130        }
131    }
132}

134/// Range mapping child bus addresses to parent bus addresses.
135///
136/// The `ranges` property uses these entries to describe how addresses
137/// on one bus are translated to another bus.
138#[derive(Clone)]
139pub struct FdtRange<'a> {
140    data_child: Raw<'a>,
141    data_parent: Raw<'a>,
142    /// Size of range
143    pub size: u64,
144}
145
146impl<'a> FdtRange<'a> {
147    /// Returns an iterator over the child bus address cells.
148    pub fn child_bus_address(&self) -> U32Iter<'a> {
149        U32Iter::new(self.data_child)
150    }
151
152    /// Returns an iterator over the parent bus address cells.
153    pub fn parent_bus_address(&self) -> U32Iter<'a> {
154        U32Iter::new(self.data_parent)
155    }
156}
157
158impl core::fmt::Debug for FdtRange<'_> {
159    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
160        f.write_str("Range {{ child_bus_address: [ ")?;
161        for addr in self.child_bus_address() {
162            f.write_fmt(format_args!("{:#x} ", addr))?;
163        }
164        f.write_str("], parent_bus_address: [ ")?;
165        for addr in self.parent_bus_address() {
166            f.write_fmt(format_args!("{:#x} ", addr))?;
167        }
168        f.write_fmt(format_args!("], size: {:#x}", self.size))
169    }
170}

172/// A slice of range entries with associated cell size information.
173#[derive(Clone)]
174pub struct FdtRangeSilce<'a> {
175    address_cell: u8,
176    address_cell_parent: u8,
177    size_cell: u8,
178    reader: Buffer<'a>,
179}
180
181impl<'a> FdtRangeSilce<'a> {
182    pub(crate) fn new(
183        address_cell: u8,
184        address_cell_parent: u8,
185        size_cell: u8,
186        raw: &Raw<'a>,
187    ) -> Self {
188        Self {
189            address_cell,
190            address_cell_parent,
191            size_cell,
192            reader: raw.buffer(),
193        }
194    }
195
196    /// Returns an iterator over the range entries.
197    pub fn iter(&self) -> FdtRangeIter<'a> {
198        FdtRangeIter { s: self.clone() }
199    }
200}
201
202/// Iterator over range entries.
203#[derive(Clone)]
204pub struct FdtRangeIter<'a> {
205    s: FdtRangeSilce<'a>,
206}
207
208impl<'a> Iterator for FdtRangeIter<'a> {
209    type Item = FdtRange<'a>;
210
211    fn next(&mut self) -> Option<Self::Item> {
212        let child_address_bytes = self.s.address_cell as usize * size_of::<u32>();
213        let data_child = self.s.reader.take(child_address_bytes).ok()?;
214
215        let parent_address_bytes = self.s.address_cell_parent as usize * size_of::<u32>();
216        let data_parent = self.s.reader.take(parent_address_bytes).ok()?;
217
218        let size = self.s.reader.take_by_cell_size(self.s.size_cell)?;
219        Some(FdtRange {
220            size,
221            data_child,
222            data_parent,
223        })
224    }
225}
