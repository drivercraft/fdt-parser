1//! FDT header structure parsing.
2//!
3//! This module handles parsing the Device Tree Blob header, which contains
4//! metadata about the structure and layout of the device tree data.
5
6use core::ptr::NonNull;
7
8use crate::FdtError;
9
10#[repr(align(4))]
11struct AlignedHeader([u8; size_of::<Header>()]);
12
13/// The FDT header structure.
14///
15/// The header is located at the start of any Device Tree Blob and contains
16/// information about the layout and version of the device tree. All multi-byte
17/// fields are stored in big-endian byte order.
18#[derive(Debug, Clone)]
19pub struct Header {
20    /// FDT header magic number (0xd00dfeed)
21    pub magic: u32,
22    /// Total size in bytes of the FDT structure
23    pub totalsize: u32,
24    /// Offset in bytes from the start of the header to the structure block
25    pub off_dt_struct: u32,
26    /// Offset in bytes from the start of the header to the strings block
27    pub off_dt_strings: u32,
28    /// Offset in bytes from the start of the header to the memory reservation block
29    pub off_mem_rsvmap: u32,
30    /// FDT version
31    pub version: u32,
32    /// Last compatible FDT version
33    pub last_comp_version: u32,
34    /// System boot CPU ID
35    pub boot_cpuid_phys: u32,
36    /// Length in bytes of the strings block
37    pub size_dt_strings: u32,
38    /// Length in bytes of the struct block
39    pub size_dt_struct: u32,
40}
41
42impl Header {
43    /// Read a header from a byte slice and return an owned `Header` whose
44    /// fields are converted from big-endian (on-disk) to host order.
45    ///
46    /// # Errors
47    ///
48    /// Returns `FdtError::BufferTooSmall` if the slice is too small.
49    /// Returns `FdtError::InvalidMagic` if the magic number is incorrect.
50    pub fn from_bytes(data: &[u8]) -> Result<Self, FdtError> {
51        if data.len() < core::mem::size_of::<Header>() {
52            return Err(FdtError::BufferTooSmall {
53                pos: core::mem::size_of::<Header>(),
54            });
55        }
56        let ptr = NonNull::new(data.as_ptr() as *mut u8).ok_or(FdtError::InvalidPtr)?;
57        unsafe { Self::from_ptr(ptr.as_ptr()) }
58    }
59
60    /// Read a header from a raw pointer and return an owned `Header` whose
61    /// fields are converted from big-endian (on-disk) to host order.
62    ///
63    /// # Safety
64    ///
65    /// The caller must ensure that the pointer is valid and points to a
66    /// memory region of at least `size_of::<Header>()` bytes that contains a
67    /// valid device tree blob.
68    pub unsafe fn from_ptr(ptr: *mut u8) -> Result<Self, FdtError> {
69        if !(ptr as usize).is_multiple_of(core::mem::align_of::<Header>()) {
70            // Pointer is not aligned, so we need to copy the data to an aligned
71            // buffer first.
72            let mut aligned = AlignedHeader([0u8; core::mem::size_of::<Header>()]);
73            unsafe {
74                core::ptr::copy_nonoverlapping(
75                    ptr,
76                    aligned.0.as_mut_ptr(),
77                    core::mem::size_of::<Header>(),
78                );
79            }
80            Self::from_aligned_ptr(aligned.0.as_mut_ptr())
81        } else {
82            // Pointer is aligned, we can read directly from it.
83            Self::from_aligned_ptr(ptr)
84        }
85    }
86
87    fn from_aligned_ptr(ptr: *mut u8) -> Result<Self, FdtError> {
88        let ptr = NonNull::new(ptr).ok_or(FdtError::InvalidPtr)?;
89
90        // SAFETY: caller provided a valid pointer to the beginning of a device
91        // tree blob. We read the raw header as it exists in memory (which is
92        // big-endian on-disk). Then convert each u32 field from big-endian to
93        // host order using `u32::from_be`.
94        let raw = unsafe { &*(ptr.cast::<Header>().as_ptr()) };
95
96        let magic = u32::from_be(raw.magic);
97        if magic != crate::FDT_MAGIC {
98            return Err(FdtError::InvalidMagic(magic));
99        }
100
101        Ok(Header {
102            magic,
103            totalsize: u32::from_be(raw.totalsize),
104            off_dt_struct: u32::from_be(raw.off_dt_struct),
105            off_dt_strings: u32::from_be(raw.off_dt_strings),
106            off_mem_rsvmap: u32::from_be(raw.off_mem_rsvmap),
107            version: u32::from_be(raw.version),
108            last_comp_version: u32::from_be(raw.last_comp_version),
109            boot_cpuid_phys: u32::from_be(raw.boot_cpuid_phys),
110            size_dt_strings: u32::from_be(raw.size_dt_strings),
111            size_dt_struct: u32::from_be(raw.size_dt_struct),
112        })
113    }
114}
