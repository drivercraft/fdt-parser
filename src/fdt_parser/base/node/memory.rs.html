1//! Memory node type.
2//!
3//! This module provides the `Memory` type for memory device nodes that
4//! describe the physical memory layout of the system.
5
6use core::{iter, ops::Deref};
7
8use crate::{base::NodeBase, FdtError, MemoryRegion};
9
10/// A memory device node.
11///
12/// Memory device nodes describe the physical memory layout for the system.
13/// A system can have multiple memory nodes, or multiple memory ranges
14/// specified in the `reg` property of a single memory node.
15#[derive(Clone)]
16pub struct Memory<'a> {
17    node: NodeBase<'a>,
18}
19
20impl<'a> Memory<'a> {
21    pub(crate) fn new(node: NodeBase<'a>) -> Self {
22        Memory { node }
23    }
24
25    /// Returns an iterator over the memory regions described by this node.
26    ///
27    /// A memory device node is required for all devicetrees and describes the
28    /// physical memory layout for the system. If a system has multiple ranges
29    /// of memory, multiple memory nodes can be created, or the ranges can be
30    /// specified in the reg property of a single memory node.
31    pub fn regions(&self) -> impl Iterator<Item = Result<MemoryRegion, FdtError>> + 'a {
32        let mut reg = self.node.reg();
33        let mut has_error = false;
34        iter::from_fn(move || {
35            if has_error {
36                return None;
37            }
38            match &mut reg {
39                Ok(iter) => {
40                    let one = iter.next()?;
41                    Some(Ok(MemoryRegion {
42                        address: one.address as usize as _,
43                        size: one.size.unwrap_or_default(),
44                    }))
45                }
46                Err(e) => {
47                    has_error = true;
48                    Some(Err(e.clone()))
49                }
50            }
51        })
52    }
53
54    /// Get the name of this memory node.
55    pub fn name(&self) -> &'a str {
56        self.node.name()
57    }
58}
59
60impl core::fmt::Debug for Memory<'_> {
61    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
62        let mut st = f.debug_struct("Memory");
63        st.field("name", &self.name());
64        for r in self.regions().flatten() {
65            st.field("region", &r);
66        }
67        st.finish()
68    }
69}
70
71impl<'a> Deref for Memory<'a> {
72    type Target = NodeBase<'a>;
73
74    fn deref(&self) -> &Self::Target {
75        &self.node
76    }
77}
