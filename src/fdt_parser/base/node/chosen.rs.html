1//! Chosen node type for boot parameters.
2//!
3//! This module provides the `Chosen` type for the /chosen node which contains
4//! system configuration parameters passed by the bootloader.
5
6use core::{fmt::Debug, ops::Deref};
7
8use crate::{base::NodeBase, FdtError};
9
10/// Result of debug console lookup.
11#[derive(Clone, Debug)]
12pub enum DebugCon<'a> {
13    /// Found the corresponding device tree node
14    Node(NodeBase<'a>),
15    /// Found earlycon parameter only in bootargs, with parsed information
16    EarlyConInfo {
17        /// The name of the early console device (e.g., "uart8250")
18        name: &'a str,
19        /// The MMIO address of the device
20        mmio: u64,
21        /// Additional parameters for the early console
22        params: Option<&'a str>,
23    },
24}
25
26/// The /chosen node containing boot parameters.
27///
28/// The chosen node doesn't represent any actual hardware device but serves
29/// as a place to pass parameters to the operating system or bootloader.
30#[derive(Clone)]
31pub struct Chosen<'a> {
32    node: NodeBase<'a>,
33}
34
35impl<'a> Chosen<'a> {
36    pub(crate) fn new(node: NodeBase<'a>) -> Self {
37        Chosen { node }
38    }
39
40    /// Get the bootargs from the bootargs property, if it exists.
41    pub fn bootargs(&self) -> Result<&'a str, FdtError> {
42        let prop = self.node.find_property("bootargs")?;
43        prop.str()
44    }
45
46    /// Get the stdout node specified by the stdout-path property.
47    ///
48    /// Searches for the node representing stdout, attempting to resolve
49    /// aliases if the node name doesn't exist as-is.
50    pub fn stdout(&self) -> Result<Stdout<'a>, FdtError> {
51        let prop = self.node.find_property("stdout-path")?;
52
53        let path = prop.str()?;
54
55        let mut sp = path.split(':');
56
57        let name = none_ok!(sp.next(), FdtError::NodeNotFound("path"));
58
59        let params = sp.next();
60        let node = self
61            .node
62            .fdt
63            .find_nodes(name)
64            .next()
65            .ok_or(FdtError::NodeNotFound("path"))??;
66
67        Ok(Stdout {
68            params,
69            node: node.deref().clone(),
70        })
71    }
72
73    /// Get the debug console information.
74    ///
75    /// First tries to find the stdout node. If that fails, parses the
76    /// bootargs for earlycon configuration.
77    pub fn debugcon(&self) -> Result<DebugCon<'a>, FdtError> {
78        match self.stdout() {
79            Ok(stdout) => Ok(DebugCon::Node(stdout.node.clone())),
80            Err(FdtError::NotFound) | Err(FdtError::NodeNotFound(_)) => {
81                self.fdt_bootargs_find_debugcon_info()
82            }
83            Err(e) => Err(e),
84        }
85    }
86
87    fn fdt_bootargs_find_debugcon_info(&self) -> Result<DebugCon<'a>, FdtError> {
88        let bootargs = self.bootargs()?;
89
90        let earlycon = none_ok!(bootargs
91            .split_ascii_whitespace()
92            .find(|&arg| arg.contains("earlycon")));
93
94        let mut tmp = earlycon.split('=');
95        let _ = none_ok!(tmp.next(), FdtError::NotFound);
96        let values = none_ok!(tmp.next(), FdtError::NotFound);
97
98        // Parse all parameters
99        let mut params_iter = values.split(',');
100        let name = none_ok!(params_iter.next(), FdtError::NotFound);
101
102        if !name.contains("uart") {
103            return Err(FdtError::NotFound);
104        }
105
106        let param2 = none_ok!(params_iter.next(), FdtError::NotFound);
107
108        let addr_str = if param2.contains("0x") {
109            param2
110        } else {
111            none_ok!(params_iter.next(), FdtError::NotFound)
112        };
113
114        let mmio = u64::from_str_radix(addr_str.trim_start_matches("0x"), 16)
115            .map_err(|_| FdtError::Utf8Parse)?;
116
117        // Try to find the corresponding node in the device tree first
118        for node_result in self.node.fdt.all_nodes() {
119            let node = node_result?;
120            match node.reg() {
121                Ok(mut regs) => {
122                    for reg in &mut regs {
123                        if reg.address == mmio {
124                            return Ok(DebugCon::Node(node.node().clone()));
125                        }
126                    }
127                }
128                Err(FdtError::NotFound) => {}
129                Err(e) => return Err(e),
130            }
131        }
132
133        // If no matching node is found, return the parsed earlycon information
134        // Re-split the string to get remaining parameters
135        let mut parts = values.split(',');
136        let _name = parts.next(); // skip name
137        let _addr_part = parts.next(); // skip address part
138        let params = if let Some(param) = parts.next() {
139            // Get the position of the first remaining parameter, then take all remaining content
140            let param_start = values.find(param).unwrap_or(0);
141            if param_start > 0 {
142                Some(&values[param_start..])
143            } else {
144                Some(param)
145            }
146        } else {
147            None
148        };
149
150        Ok(DebugCon::EarlyConInfo { name, mmio, params })
151    }
152}

154impl Debug for Chosen<'_> {
155    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
156        f.debug_struct("Chosen")
157            .field("bootargs", &self.bootargs())
158            .field("stdout", &self.stdout())
159            .finish()
160    }
161}
162
163impl<'a> Deref for Chosen<'a> {
164    type Target = NodeBase<'a>;
165
166    fn deref(&self) -> &Self::Target {
167        &self.node
168    }
169}

171/// The stdout device specified by the chosen node.
172///
173/// Contains the node reference and optional parameters (typically specifying
174/// the baud rate or other console configuration).
175#[derive(Clone)]
176pub struct Stdout<'a> {
177    /// Optional parameters for the stdout device (e.g., baud rate)
178    pub params: Option<&'a str>,
179    /// The device tree node for the stdout device
180    pub node: NodeBase<'a>,
181}
182
183impl<'a> Stdout<'a> {}
184
185impl Debug for Stdout<'_> {
186    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
187        f.debug_struct("Stdout")
188            .field("name", &self.node.name())
189            .field("params", &self.params)
190            .finish()
191    }
192}
193
194impl<'a> Deref for Stdout<'a> {
195    type Target = NodeBase<'a>;
196
197    fn deref(&self) -> &Self::Target {
198        &self.node
199    }
