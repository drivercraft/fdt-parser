25/// Base node type representing any device tree node.
26///
27/// `NodeBase` provides common functionality available on all nodes,
28/// including property access, child iteration, and parent references.
29#[derive(Clone)]
30pub struct NodeBase<'a> {
31    name: &'a str,
32    pub(crate) fdt: Fdt<'a>,
33    /// The depth/level of this node in the device tree (0 for root)
34    pub level: usize,
35    pub(crate) raw: Raw<'a>,
36    pub(crate) parent: Option<ParentInfo<'a>>,
37    interrupt_parent: Option<Phandle>,
38}
39
40/// Information about a node's parent, used for address translation.
41#[derive(Clone)]
42pub(crate) struct ParentInfo<'a> {
43    pub name: &'a str,
44    pub level: usize,
45    pub raw: Raw<'a>,
46    /// Parent's #address-cells and #size-cells (for parsing reg)
47    pub address_cells: Option<u8>,
48    pub size_cells: Option<u8>,
49    /// Parent's ranges for address translation
50    pub ranges: Option<FdtRangeSilce<'a>>,
51}

53/// Builder for creating NodeBase with parent information.
54///
55/// This struct reduces the number of parameters needed for `NodeBase::new_with_parent_info`
56/// by grouping related parameters together.
57pub(crate) struct ParentInfoBuilder<'a> {
58    pub parent_address_cells: Option<u8>,
59    pub parent_size_cells: Option<u8>,
60    pub parent_ranges: Option<FdtRangeSilce<'a>>,
61    pub interrupt_parent: Option<Phandle>,
62}

64impl<'a> NodeBase<'a> {
65    /// Create a new NodeBase with pre-calculated parent information from the stack.
66    #[allow(clippy::too_many_arguments)]
67    pub(crate) fn new_with_parent_info(
68        name: &'a str,
69        fdt: Fdt<'a>,
70        raw: Raw<'a>,
71        level: usize,
72        parent: Option<&NodeBase<'a>>,
73        parent_info: ParentInfoBuilder<'a>,
74    ) -> Self {
75        let name = if name.is_empty() { "/" } else { name };
76        NodeBase {
77            name,
78            fdt,
79            level,
80            parent: parent.map(|p| ParentInfo {
81                name: p.name(),
82                level: p.level(),
83                raw: p.raw(),
84                address_cells: parent_info.parent_address_cells,
85                size_cells: parent_info.parent_size_cells,
86                ranges: parent_info.parent_ranges,
87            }),
88            interrupt_parent: parent_info.interrupt_parent,
89            raw,
90        }
91    }
92
93    /// Returns the name of this node's parent.
94    pub fn parent_name(&self) -> Option<&'a str> {
95        self.parent_fast().map(|p| p.name())
96    }
97
98    /// Returns the parent node as a `Node`.
99    pub fn parent(&self) -> Option<Node<'a>> {
100        let parent_info = self.parent.as_ref()?;
101        self.fdt
102            .all_nodes()
103            .flatten()
104            .find(|node| node.name() == parent_info.name && node.level() == parent_info.level)
105    }
106
107    pub(crate) fn parent_fast(&self) -> Option<NodeBase<'a>> {
108        self.parent.as_ref().map(|p| NodeBase {
109            name: p.name,
110            fdt: self.fdt.clone(),
111            level: p.level,
112            raw: p.raw,
113            parent: None,
114            interrupt_parent: None,
115        })
116    }
117
118    /// Returns the raw data for this node.
119    pub fn raw(&self) -> Raw<'a> {
120        self.raw
121    }
122
123    /// Get the name of this node.
124    pub fn name(&self) -> &'a str {
125        self.name
126    }
127
128    /// Get the level/depth of this node in the device tree.
129    pub fn level(&self) -> usize {
130        self.level
131    }
132
133    /// Get compatible strings for this node (placeholder implementation).
134    pub fn compatibles(&self) -> Result<impl Iterator<Item = &'a str> + 'a, FdtError> {
135        let prop = self.find_property("compatible")?;
136        Ok(prop.str_list())
137    }
138
139    /// Returns a flattened iterator over compatible strings.
140    ///
141    /// This is an alias for [`compatibles`](Self::compatibles) that
142    /// returns the same iterator for chaining with other iterator operations.
143    pub fn compatibles_flatten(&self) -> Result<impl Iterator<Item = &'a str> + 'a, FdtError> {
144        self.compatibles()
145    }
146
147    /// Returns an iterator over this node's register entries.
148    ///
149    /// The addresses are automatically translated from child bus addresses
150    /// to parent bus addresses using the parent's ranges property.
151    pub fn reg(&self) -> Result<RegIter<'a>, FdtError> {
152        let prop = self.find_property("reg")?;
153
154        // Get parent info from ParentInfo structure
155        let parent_info = self
156            .parent
157            .as_ref()
158            .ok_or(FdtError::NodeNotFound("parent"))?;
159
160        // reg parsing uses the immediate parent's cells
161        let address_cell = parent_info.address_cells.unwrap_or(2);
162        let size_cell = parent_info.size_cells.unwrap_or(1);
163
164        // Use parent's pre-calculated ranges for address translation
165        let ranges = parent_info.ranges.clone();
166
167        Ok(RegIter {
168            size_cell,
169            address_cell,
170            buff: prop.data.buffer(),
171            ranges,
172        })
173    }
174
175    fn is_interrupt_controller(&self) -> bool {
176        self.find_property("#interrupt-controller").is_ok()
177    }
178
179    /// Check if this node is the root node.
180    pub fn is_root(&self) -> bool {
181        self.level == 0
182    }
183
184    /// Get debug information about the node (for debugging purposes only).
185    pub fn debug_info(&self) -> NodeDebugInfo<'a> {
186        NodeDebugInfo {
187            name: self.name(),
188            level: self.level,
189            pos: self.raw.pos(),
190        }
191    }
192
193    /// Returns an iterator over this node's properties.
194    pub fn properties(&self) -> impl Iterator<Item = Result<Property<'a>, FdtError>> + '_ {
195        let reader = self.raw.buffer();
196        PropIter::new(self.fdt.clone(), reader)
197    }
198
199    /// Find a property by name.
200    pub fn find_property(&self, name: &str) -> Result<Property<'a>, FdtError> {
201        for prop in self.properties() {
202            let prop = prop?;
203            if prop.name.eq(name) {
204                return Ok(prop);
205            }
206        }
207        Err(FdtError::NotFound)
208    }
209
210    /// Get this node's phandle.
211    pub fn phandle(&self) -> Result<Phandle, FdtError> {
212        let prop = self.find_property("phandle")?;
213        Ok(prop.u32()?.into())
214    }
215
216    /// Find [InterruptController] from current node or its parent.
217    pub fn interrupt_parent(&self) -> Result<InterruptController<'a>, FdtError> {
218        // First try to get the interrupt parent phandle from the node itself
219        let phandle = self.interrupt_parent.ok_or(FdtError::NotFound)?;
220
221        // Find the node with this phandle
222        let node = self.fdt.get_node_by_phandle(phandle)?;
223        match node {
224            Node::InterruptController(ic) => Ok(ic),
225            _ => Err(FdtError::NodeNotFound("interrupt-parent")),
226        }
227    }
228
229    /// Get the interrupt parent phandle for this node.
230    pub fn get_interrupt_parent_phandle(&self) -> Option<Phandle> {
231        self.interrupt_parent
232    }
233
234    /// Returns an iterator over this node's interrupts.
235    ///
236    /// Each interrupt is represented as an iterator of u32 cells.
237    pub fn interrupts(
238        &self,
239    ) -> Result<impl Iterator<Item = impl Iterator<Item = u32> + 'a> + 'a, FdtError> {
240        let prop = self.find_property("interrupts")?;
241        let irq_parent = self.interrupt_parent()?;
242        let cell_size = irq_parent.interrupt_cells()?;
243        let iter = U32Iter2D::new(&prop.data, cell_size);
244
245        Ok(iter)
246    }
247
248    /// Get the clock-frequency property value.
249    pub fn clock_frequency(&self) -> Result<u32, FdtError> {
250        let prop = self.find_property("clock-frequency")?;
251        prop.u32()
252    }
253
254    /// Returns an iterator over this node's children.
255    pub fn children(&self) -> NodeChildIter<'a> {
256        NodeChildIter {
257            fdt: self.fdt.clone(),
258            parent: self.clone(),
259            all_nodes: None,
260            target_level: 0,
261            found_parent: false,
262        }
263    }
264
265    /// Get the status property value.
266    pub fn status(&self) -> Result<Status, FdtError> {
267        let prop = self.find_property("status")?;
268        let s = prop.str()?;
269
270        if s.contains("disabled") {
271            return Ok(Status::Disabled);
272        }
273
274        if s.contains("okay") {
275            return Ok(Status::Okay);
276        }
277
278        Err(FdtError::NotFound)
279    }

299/// Iterator over register entries.
300pub struct RegIter<'a> {
301    pub(crate) size_cell: u8,
302    pub(crate) address_cell: u8,
303    pub(crate) buff: Buffer<'a>,
304    pub(crate) ranges: Option<FdtRangeSilce<'a>>,
305}
306impl Iterator for RegIter<'_> {
307    type Item = FdtReg;
308
309    fn next(&mut self) -> Option<Self::Item> {
310        let child_bus_address = self.buff.take_by_cell_size(self.address_cell)?;
311
312        let mut address = child_bus_address;
313
314        if let Some(ranges) = &self.ranges {
315            for one in ranges.iter() {
316                let range_child_bus_address = one.child_bus_address().as_u64();
317                let range_parent_bus_address = one.parent_bus_address().as_u64();
318
319                if child_bus_address >= range_child_bus_address
320                    && child_bus_address < range_child_bus_address + one.size
321                {
322                    address =
323                        child_bus_address - range_child_bus_address + range_parent_bus_address;
324                    break;
325                }
326            }
327        }
328
329        let size = if self.size_cell > 0 {
330            Some(self.buff.take_by_cell_size(self.size_cell)? as usize)
331        } else {
332            None
333        };
334        Some(FdtReg {
335            address,
336            child_bus_address,
337            size,
338        })
339    }
340}

342/// Typed node enum for specialized node access.
343///
344/// Nodes are automatically classified based on their name and properties.
345/// Use pattern matching to access node-specific functionality.
346#[derive(Debug, Clone)]
347pub enum Node<'a> {
348    /// A general-purpose node without special handling
349    General(NodeBase<'a>),
350    /// The /chosen node containing boot parameters
351    Chosen(Chosen<'a>),
352    /// A memory node (e.g., /memory@0)
353    Memory(Memory<'a>),
354    /// An interrupt controller node
355    InterruptController(InterruptController<'a>),
356}
357
358impl<'a> Node<'a> {
359    /// Returns a reference to the underlying `NodeBase`.
360    pub fn node(&self) -> &NodeBase<'a> {
361        self.deref()
362    }
363}
364
365impl<'a> From<NodeBase<'a>> for Node<'a> {
366    fn from(node: NodeBase<'a>) -> Self {
367        if node.name() == "chosen" {
368            Node::Chosen(Chosen::new(node))
369        } else if node.name().starts_with("memory@") {
370            Node::Memory(Memory::new(node))
371        } else if node.is_interrupt_controller() {
372            Node::InterruptController(InterruptController::new(node))
373        } else {
374            Node::General(node)
375        }
376    }
377}
378
379impl<'a> Deref for Node<'a> {
380    type Target = NodeBase<'a>;
381
382    fn deref(&self) -> &Self::Target {
383        match self {
384            Node::General(n) => n,
385            Node::Chosen(n) => n,
386            Node::Memory(n) => n,
387            Node::InterruptController(n) => n,
388        }
389    }
390}

392/// Iterator over a node's children.
393pub struct NodeChildIter<'a> {
394    fdt: Fdt<'a>,
395    parent: NodeBase<'a>,
396    all_nodes: Option<NodeIter<'a, 16>>,
397    target_level: usize,
398    found_parent: bool,
399}
400
401impl<'a> Iterator for NodeChildIter<'a> {
402    type Item = Result<Node<'a>, FdtError>;
403
404    fn next(&mut self) -> Option<Self::Item> {
405        // Lazily initialize the node iterator
406        if self.all_nodes.is_none() {
407            self.all_nodes = Some(self.fdt.all_nodes());
408        }
409
410        let all_nodes = self.all_nodes.as_mut()?;
411
412        // Search for child nodes
413        loop {
414            let node = match all_nodes.next()? {
415                Ok(node) => node,
416                Err(e) => return Some(Err(e)),
417            };
418
419            // First, find the parent node
420            if !self.found_parent {
421                if node.name() == self.parent.name() && node.level() == self.parent.level() {
422                    self.found_parent = true;
423                    self.target_level = node.level() + 1;
424                }
425                continue;
426            }
427
428            // Parent node found, now look for child nodes
429            let current_level = node.level();
430
431            // If current node's level equals target level and follows parent in tree structure,
432            // then it's a direct child of the parent node
433            if current_level == self.target_level {
434                return Some(Ok(node));
435            }
436
437            // If current node's level is less than or equal to parent's level,
438            // we've left the parent's subtree
439            if current_level <= self.parent.level() {
440                break;
441            }
442        }
443
444        None
445    }
446}
447
448impl<'a> NodeChildIter<'a> {
449    /// Create a new child node iterator.
450    pub fn new(fdt: Fdt<'a>, parent: NodeBase<'a>) -> Self {
451        NodeChildIter {
452            fdt,
453            parent,
454            all_nodes: None,
455            target_level: 0,
456            found_parent: false,
457        }
458    }
459
460    /// Get a reference to the parent node.
461    pub fn parent(&self) -> &NodeBase<'a> {
462        &self.parent
463    }
464
465    /// Collect all child nodes into a Vec.
466    pub fn collect_children(self) -> Result<alloc::vec::Vec<Node<'a>>, FdtError> {
467        self.collect()
468    }
469
470    /// Find a child node by name.
471    pub fn find_child_by_name(self, name: &str) -> Result<Node<'a>, FdtError> {
472        for child_result in self {
473            let child = child_result?;
474            if child.name() == name {
475                return Ok(child);
476            }
477        }
478        Err(FdtError::NotFound)
479    }
480
481    /// Find a child node by compatible string.
482    pub fn find_child_by_compatible(self, compatible: &str) -> Result<Node<'a>, FdtError> {
483        for child_result in self {
484            let child = child_result?;
485            match child.compatibles() {
486                Ok(mut compatibles) => {
487                    if compatibles.any(|comp| comp == compatible) {
488                        return Ok(child);
489                    }
490                }
491                Err(FdtError::NotFound) => {}
492                Err(e) => return Err(e),
493            }
494        }
495        Err(FdtError::NotFound)
496    }
497}
