1//! Low-level data access utilities for FDT parsing.
2//!
3//! This module provides raw data access primitives for reading and parsing
4//! Device Tree Blob data structures. It handles byte-aligned access, buffer
5//! management, and iterators for common data formats used in device trees.
6
7use core::{
8    ffi::CStr,
9    ops::{Deref, Range},
10};

12use crate::{base::Fdt, FdtError, Property, Token};

14/// A raw byte slice view with position tracking for FDT data.
15///
16/// `Raw` provides a window into the FDT data with the ability to track
17/// the current position and create sub-ranges.
18#[derive(Clone, Copy)]
19pub struct Raw<'a> {
20    value: &'a [u8],
21    pos: usize,
22}
23
24impl<'a> Raw<'a> {
25    /// Creates a new `Raw` view from a byte slice.
26    pub(crate) fn new(value: &'a [u8]) -> Self {
27        Raw { value, pos: 0 }
28    }
29
30    /// Creates a new `Buffer` for sequential reading from this raw data.
31    pub fn buffer(&self) -> Buffer<'a> {
32        Buffer {
33            raw: *self,
34            iter: 0,
35        }
36    }
37
38    /// Returns the underlying byte slice.
39    pub fn value(&self) -> &'a [u8] {
40        self.value
41    }
42
43    /// Creates a new `Raw` starting at the specified offset from the current position.
44    pub fn begin_at(&self, offset: usize) -> Raw<'a> {
45        let pos = self.pos + offset;
46        Raw {
47            value: &self.value[offset..],
48            pos,
49        }
50    }
51
52    /// Returns a sub-range of the data as a new `Raw`.
53    ///
54    /// # Errors
55    ///
56    /// Returns `FdtError::BufferTooSmall` if the range extends beyond the data.
57    pub fn get_range(&self, range: Range<usize>) -> Result<Raw<'a>, FdtError> {
58        let pos = self.pos + range.start;
59        let end = pos + range.len();
60        if end <= self.value.len() {
61            Ok(Raw {
62                value: &self.value[range],
63                pos,
64            })
65        } else {
66            Err(FdtError::BufferTooSmall { pos: end })
67        }
68    }
69
70    /// Returns the current position in the original data stream.
71    pub fn pos(&self) -> usize {
72        self.pos
73    }
74
75    /// Returns the underlying byte slice as a reference.
76    pub fn as_ref(&self) -> &'a [u8] {
77        self.value
78    }
79}
80
81impl<'a> Deref for Raw<'a> {
82    type Target = &'a [u8];
83
84    fn deref(&self) -> &Self::Target {
85        &self.value
86    }
87}

89/// A sequential buffer reader for parsing FDT data structures.
90///
91/// `Buffer` provides sequential read access with automatic position tracking,
92/// supporting various data types and alignment operations required by the
93/// Device Tree specification.
94#[derive(Clone)]
95pub struct Buffer<'a> {
96    raw: Raw<'a>,
97    iter: usize,
98}
99
100impl<'a> Buffer<'a> {
101    /// Takes the specified number of bytes from the buffer.
102    ///
103    /// # Errors
104    ///
105    /// Returns `FdtError::BufferTooSmall` if insufficient bytes remain.
106    pub fn take(&mut self, size: usize) -> Result<Raw<'a>, FdtError> {
107        let start = self.iter;
108        let end = start + size;
109        if end <= self.raw.value.len() {
110            self.iter = end;
111            Ok(Raw {
112                value: &self.raw.value[start..end],
113                pos: self.pos(),
114            })
115        } else {
116            Err(FdtError::BufferTooSmall {
117                pos: self.pos() + size,
118            })
119        }
120    }
121
122    pub(crate) fn pos(&self) -> usize {
123        self.raw.pos + self.iter
124    }
125
126    /// Returns the remaining unread data as a `Raw`.
127    pub fn remain(&self) -> Raw<'a> {
128        Raw {
129            value: &self.raw.value[self.iter..],
130            pos: self.pos(),
131        }
132    }
133
134    /// Reads a big-endian u32 value.
135    pub fn take_u32(&mut self) -> Result<u32, FdtError> {
136        let bytes = self.take(4)?;
137        Ok(u32::from_be_bytes(bytes.as_ref().try_into().unwrap()))
138    }
139
140    /// Reads a big-endian u64 value.
141    pub fn take_u64(&mut self) -> Result<u64, FdtError> {
142        let bytes = self.take(8)?;
143        Ok(u64::from_be_bytes(bytes.as_ref().try_into().unwrap()))
144    }
145
146    pub(crate) fn take_token(&mut self) -> Result<Token, FdtError> {
147        let u = self.take_u32()?;
148        Ok(Token::from(u))
149    }
150
151    /// Reads a null-terminated string.
152    pub fn take_str(&mut self) -> Result<&'a str, FdtError> {
153        let remain = self.remain();
154        if remain.is_empty() {
155            return Err(FdtError::BufferTooSmall { pos: self.iter });
156        }
157
158        let cs = CStr::from_bytes_until_nul(remain.as_ref())
159            .map_err(|_| FdtError::FromBytesUntilNull)?;
160
161        let s = cs.to_str()?;
162
163        let str_len = cs.to_bytes_with_nul().len();
164        self.iter += str_len;
165
166        Ok(s)
167    }
168
169    /// Skips bytes aligned to 4-byte boundary (FDT format requirement).
170    pub fn skip_4_aligned(&mut self, len: usize) -> Result<(), FdtError> {
171        self.take((len + 3) & !0x3)?;
172        Ok(())
173    }
174
175    /// Takes bytes aligned to 4-byte boundary.
176    pub fn take_aligned(&mut self, len: usize) -> Result<Raw<'a>, FdtError> {
177        let bytes = (len + 3) & !0x3;
178        self.take(bytes)
179    }
180
181    /// Advances the position to the next 4-byte boundary.
182    pub fn take_to_aligned(&mut self) {
183        let remain = self.iter % 4;
184        if remain != 0 {
185            let add = 4 - remain;
186            if self.iter + add <= self.raw.value.len() {
187                self.iter += 4 - remain;
188            } else {
189                self.iter = self.raw.value.len();
190            }
191        }
192    }
193
194    /// Takes a value based on the cell size (1 = 4 bytes, 2 = 8 bytes).
195    ///
196    /// # Panics
197    ///
198    /// Panics if cell_size is not 1 or 2.
199    pub fn take_by_cell_size(&mut self, cell_size: u8) -> Option<u64> {
200        match cell_size {
201            1 => self.take_u32().map(|s| s as _).ok(),
202            2 => self.take_u64().ok(),
203            _ => panic!("invalid cell size {}", cell_size),
204        }
205    }
206
207    /// Takes a property value from the buffer.
208    pub fn take_prop(&mut self, fdt: &Fdt<'a>) -> Result<Property<'a>, FdtError> {
209        let len = self.take_u32()?;
210        let nameoff = self.take_u32()?;
211        let data = self.take_aligned(len as _)?;
212        Ok(Property {
213            name: fdt.get_str(nameoff as _)?,
214            data,
215        })
216    }
217}

219/// Iterator over u32 values in raw data.
220pub struct U32Iter<'a> {
221    buffer: Buffer<'a>,
222}
223
224impl<'a> U32Iter<'a> {
225    /// Creates a new u32 iterator from raw data.
226    pub fn new(raw: Raw<'a>) -> Self {
227        Self {
228            buffer: raw.buffer(),
229        }
230    }
231
232    /// Reads two u32 values as a u64 (big-endian combination).
233    pub fn as_u64(&mut self) -> u64 {
234        let h = self.buffer.take_u32().unwrap();
235        if let Ok(l) = self.buffer.take_u32() {
236            ((h as u64) << 32) + l as u64
237        } else {
238            h as _
239        }
240    }
241}
242
243impl<'a> Iterator for U32Iter<'a> {
244    type Item = u32;
245
246    fn next(&mut self) -> Option<Self::Item> {
247        self.buffer.take_u32().ok()
248    }
249}

251/// Iterator over 2D arrays of u32 values.
252pub struct U32Iter2D<'a> {
253    reader: Buffer<'a>,
254    row_len: u8,
255}
256
257impl<'a> U32Iter2D<'a> {
258    /// Creates a new 2D iterator with the specified row length (in u32 cells).
259    pub fn new(bytes: &Raw<'a>, row_len: u8) -> Self {
260        Self {
261            reader: bytes.buffer(),
262            row_len,
263        }
264    }
265}
266
267impl<'a> Iterator for U32Iter2D<'a> {
268    type Item = U32Iter<'a>;
269
270    fn next(&mut self) -> Option<Self::Item> {
271        let bytes = self
272            .reader
273            .take(self.row_len as usize * size_of::<u32>())
274            .ok()?;
275        Some(U32Iter::new(bytes))
276    }
