14/// Iterator over memory reservation entries.
15///
16/// The memory reservation block contains a list of physical memory regions
17/// that must be preserved during boot. This iterator yields each reservation
18/// entry until it reaches the terminating entry (address=0, size=0).
19pub struct MemoryReservationIter<'a> {
20    data: &'a [u8],
21    offset: usize,
22}
23
24impl<'a> Iterator for MemoryReservationIter<'a> {
25    type Item = MemoryReservation;
26
27    fn next(&mut self) -> Option<Self::Item> {
28        // Ensure we have enough data to read address and size (8 bytes each)
29        if self.offset + 16 > self.data.len() {
30            return None;
31        }
32
33        // Read address (8 bytes, big-endian)
34        let address_bytes = &self.data[self.offset..self.offset + 8];
35        let address = u64::from_be_bytes(address_bytes.try_into().unwrap());
36        self.offset += 8;
37
38        // Read size (8 bytes, big-endian)
39        let size_bytes = &self.data[self.offset..self.offset + 8];
40        let size = u64::from_be_bytes(size_bytes.try_into().unwrap());
41        self.offset += 8;
42
43        // Check for terminator (both address and size are zero)
44        if address == 0 && size == 0 {
45            return None;
46        }
47
48        Some(MemoryReservation { address, size })
49    }
50}

52/// Helper function for writing indentation during formatting.
53fn write_indent(f: &mut fmt::Formatter<'_>, count: usize, ch: &str) -> fmt::Result {
54    for _ in 0..count {
55        write!(f, "{}", ch)?;
56    }
57    Ok(())

60/// A parsed Flattened Device Tree (FDT).
61///
62/// This is the main type for working with device tree blobs. It provides
63/// methods for traversing the tree, finding nodes by path, translating
64/// addresses, and accessing special nodes like /chosen and /memory.
65///
66/// The `Fdt` holds a reference to the underlying device tree data and
67/// performs zero-copy parsing where possible.
68#[derive(Clone)]
69pub struct Fdt<'a> {
70    header: Header,
71    pub(crate) data: Bytes<'a>,
72}
73
74impl<'a> Fdt<'a> {
75    /// Create a new `Fdt` from a byte slice.
76    ///
77    /// Parses the FDT header and validates the magic number. The slice
78    /// must contain a complete, valid device tree blob.
79    ///
80    /// # Errors
81    ///
82    /// Returns `FdtError` if the header is invalid, the magic number
83    /// doesn't match, or the buffer is too small.
84    pub fn from_bytes(data: &'a [u8]) -> Result<Fdt<'a>, FdtError> {
85        let header = Header::from_bytes(data)?;
86        if data.len() < header.totalsize as usize {
87            return Err(FdtError::BufferTooSmall {
88                pos: header.totalsize as usize,
89            });
90        }
91        let buffer = Bytes::new(data);
92
93        Ok(Fdt {
94            header,
95            data: buffer,
96        })
97    }
98
99    /// Create a new `Fdt` from a raw pointer.
100    ///
101    /// Parses an FDT from the memory location pointed to by `ptr`.
102    /// This is useful when working with device trees loaded by bootloaders.
103    ///
104    /// # Safety
105    ///
106    /// The caller must ensure that the pointer is valid and points to a
107    /// memory region of at least `totalsize` bytes that contains a valid
108    /// device tree blob. The memory must remain valid for the lifetime `'a`.
109    ///
110    /// # Errors
111    ///
112    /// Returns `FdtError` if the header is invalid or the magic number
113    /// doesn't match.
114    pub unsafe fn from_ptr(ptr: *mut u8) -> Result<Fdt<'a>, FdtError> {
115        let header = unsafe { Header::from_ptr(ptr)? };
116
117        let data_slice = unsafe { core::slice::from_raw_parts(ptr, header.totalsize as _) };
118        let data = Bytes::new(data_slice);
119
120        Ok(Fdt { header, data })
121    }

126    }

138    /// Find a node by its absolute path or alias.
139    ///
140    /// The path can be an absolute path starting with '/', or an alias
141    /// defined in the /aliases node. Returns `None` if the node is not found.
142    ///
143    /// # Example
144    ///
145    /// ```ignore
146    /// let node = fdt.find_by_path("/soc@30000000/serial@10000");
147    /// let uart = fdt.find_by_path("serial0");  // Using alias
148    /// ```
149    pub fn find_by_path(&self, path: &str) -> Option<Node<'a>> {
150        let path = self.normalize_path(path)?;
151        let split = path.trim_matches('/').split('/');
152
153        let mut current_iter = self.all_nodes();
154        let mut found_node: Option<Node<'a>> = None;
155
156        for part in split {
157            let mut found = false;
158            for node in current_iter.by_ref() {
159                let node_name = node.name();
160                if node_name == part {
161                    found = true;
162                    found_node = Some(node);
163                    break;
164                }
165            }
166            if !found {
167                return None;
168            }
169        }
170
171        found_node
172    }

174    /// Resolve an alias to its full path.
175    ///
176    /// Looks up the alias in the /aliases node and returns the corresponding
177    /// path string.
178    fn resolve_alias(&self, alias: &str) -> Option<&'a str> {
179        let aliases_node = self.find_by_path("/aliases")?;
180        aliases_node.find_property_str(alias)
181    }
182
183    /// Normalize a path to an absolute path.
184    ///
185    /// If the path starts with '/', it's returned as-is. Otherwise,
186    /// it's treated as an alias and resolved.
187    fn normalize_path(&self, path: &'a str) -> Option<&'a str> {
188        if path.starts_with('/') {
189            Some(path)
190        } else {
191            self.resolve_alias(path)
192        }
193    }

195    /// Translate a device address to a CPU physical address.
196    ///
197    /// This function implements address translation similar to Linux's
198    /// `of_translate_address`. It walks up the device tree hierarchy,
199    /// applying each parent's `ranges` property to translate the child
200    /// address space to the parent address space.
201    ///
202    /// The translation starts from the node at `path` and walks up through
203    /// each parent, applying the `ranges` property until reaching the root.
204    ///
205    /// # Arguments
206    ///
207    /// * `path` - Node path (absolute path starting with '/' or alias name)
208    /// * `address` - Device address from the node's `reg` property
209    ///
210    /// # Returns
211    ///
212    /// The translated CPU physical address. If translation fails at any
213    /// point (e.g., a parent node has no `ranges` property), the original
214    /// address is returned.
215    pub fn translate_address(&self, path: &'a str, address: u64) -> u64 {
216        let path = match self.normalize_path(path) {
217            Some(p) => p,
218            None => return address,
219        };
220
221        // Split path into component parts
222        let path_parts: heapless::Vec<&str, 16> = path
223            .trim_matches('/')
224            .split('/')
225            .filter(|s| !s.is_empty())
226            .collect();
227
228        if path_parts.is_empty() {
229            return address;
230        }
231
232        let mut current_address = address;
233
234        // Walk up from the deepest node, applying ranges at each level
235        // Note: We start from the second-to-last level (the target node itself is skipped)
236        for depth in (0..path_parts.len()).rev() {
237            // Build the path to the current parent level
238            let parent_parts = &path_parts[..depth];
239            if parent_parts.is_empty() {
240                // Reached root node, no more translation needed
241                break;
242            }
243
244            // Find the parent node
245            let mut parent_path = heapless::String::<256>::new();
246            parent_path.push('/').ok();
247            for (i, part) in parent_parts.iter().enumerate() {
248                if i > 0 {
249                    parent_path.push('/').ok();
250                }
251                parent_path.push_str(part).ok();
252            }
253
254            let parent_node = match self.find_by_path(parent_path.as_str()) {
255                Some(node) => node,
256                None => continue,
257            };
258
259            // Get the parent's ranges property
260            let ranges = match parent_node.ranges() {
261                Some(r) => r,
262                None => {
263                    // No ranges property, stop translation
264                    break;
265                }
266            };
267
268            // Look for a matching translation rule in ranges
269            let mut found = false;
270            for range in ranges.iter() {
271                // Check if the address falls within this range
272                if current_address >= range.child_address
273                    && current_address < range.child_address + range.length
274                {
275                    // Calculate offset in child address space
276                    let offset = current_address - range.child_address;
277                    // Translate to parent address space
278                    current_address = range.parent_address + offset;
279                    found = true;
280                    break;
281                }
282            }
283
284            if !found {
285                // No matching range found, keep current address and continue
286                // This typically means translation failed, but we try upper levels
287            }
288        }
289
290        current_address
291    }

301    /// Returns the /chosen node if it exists.
302    pub fn chosen(&self) -> Option<Chosen<'a>> {
303        for node in self.all_nodes() {
304            if let Node::Chosen(c) = node {
305                return Some(c);
306            }
307        }

311    /// Returns an iterator over all memory nodes.
312    pub fn memory(&self) -> impl Iterator<Item = Memory<'a>> + 'a {
313        self.all_nodes().filter_map(|node| {
314            if let Node::Memory(mem) = node {
315                Some(mem)
316            } else {
317                None

332/// Iterator over nodes in the /reserved-memory region.
333///
334/// Yields all child nodes of the /reserved-memory node, which describe
335/// memory regions that are reserved for specific purposes.
336struct ReservedMemoryIter<'a> {
337    node_iter: FdtIter<'a>,
338    in_reserved_memory: bool,
339    reserved_level: usize,
340}
341
342impl<'a> Iterator for ReservedMemoryIter<'a> {
343    type Item = Node<'a>;
344
345    fn next(&mut self) -> Option<Self::Item> {
346        for node in self.node_iter.by_ref() {
347            if node.name() == "reserved-memory" {
348                self.in_reserved_memory = true;
349                self.reserved_level = node.level();
350                continue;
351            }
352
353            if self.in_reserved_memory {
354                if node.level() <= self.reserved_level {
355                    // Left the reserved-memory node
356                    self.in_reserved_memory = false;
357                    return None;
358                } else {
359                    return Some(node);
360                }
361            }
362        }
363        None
364    }
365}

367impl fmt::Display for Fdt<'_> {
368    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
369        writeln!(f, "/dts-v1/;")?;
370        writeln!(f)?;
371
372        let mut prev_level = 0;
373
374        for node in self.all_nodes() {
375            let level = node.level();
376
377            // Close nodes from the previous level
378            while prev_level > level {
379                prev_level -= 1;
380                write_indent(f, prev_level, "    ")?;
381                writeln!(f, "}};\n")?;
382            }
383
384            write_indent(f, level, "    ")?;
385            let name = if node.name().is_empty() {
386                "/"
387            } else {
388                node.name()
389            };
390
391            // Print node header
392            writeln!(f, "{} {{", name)?;
393
394            // Print properties
395            for prop in node.properties() {
396                write_indent(f, level + 1, "    ")?;
397                writeln!(f, "{};", prop)?;
398            }
399
400            prev_level = level + 1;
401        }
402
403        // Close remaining nodes
404        while prev_level > 0 {
405            prev_level -= 1;
406            write_indent(f, prev_level, "    ")?;
407            writeln!(f, "}};\n")?;
408        }
409
410        Ok(())
411    }
412}

414impl fmt::Debug for Fdt<'_> {
415    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
416        writeln!(f, "Fdt {{")?;
417        writeln!(f, "\theader: {:?}", self.header)?;
418        writeln!(f, "\tnodes:")?;
419
420        for node in self.all_nodes() {
421            let level = node.level();
422            // Base indentation is 2 tabs, plus 1 tab per level
423            write_indent(f, level + 2, "\t")?;
424
425            let name = if node.name().is_empty() {
426                "/"
427            } else {
428                node.name()
429            };
430
431            // Print node name and basic info
432            writeln!(
433                f,
434                "[{}] address_cells={}, size_cells={}",
435                name, node.address_cells, node.size_cells
436            )?;
437
438            // Print properties
439            for prop in node.properties() {
440                write_indent(f, level + 3, "\t")?;
441                if let Some(v) = prop.as_address_cells() {
442                    writeln!(f, "#address-cells: {}", v)?;
443                } else if let Some(v) = prop.as_size_cells() {
444                    writeln!(f, "#size-cells: {}", v)?;
445                } else if let Some(v) = prop.as_interrupt_cells() {
446                    writeln!(f, "#interrupt-cells: {}", v)?;
447                } else if let Some(s) = prop.as_status() {
448                    writeln!(f, "status: {:?}", s)?;
449                } else if let Some(p) = prop.as_phandle() {
450                    writeln!(f, "phandle: {}", p)?;
451                } else {
452                    // Default handling for unknown properties
453                    if prop.is_empty() {
454                        writeln!(f, "{}", prop.name())?;
455                    } else if let Some(s) = prop.as_str() {
456                        writeln!(f, "{}: \"{}\"", prop.name(), s)?;
457                    } else if prop.len() == 4 {
458                        let v = u32::from_be_bytes(prop.data().as_slice().try_into().unwrap());
459                        writeln!(f, "{}: {:#x}", prop.name(), v)?;
460                    } else {
461                        writeln!(f, "{}: <{} bytes>", prop.name(), prop.len())?;
462                    }
463                }
464            }
465        }
466
467        writeln!(f, "}}")
468    }

476    #[test]
477    fn test_memory_reservation_iterator() {
478        // Create simple test data: one memory reservation entry + terminator
479        let mut test_data = [0u8; 32];
480
481        // Address: 0x80000000, Size: 0x10000000 (256MB)
482        test_data[0..8].copy_from_slice(&0x80000000u64.to_be_bytes());
483        test_data[8..16].copy_from_slice(&0x10000000u64.to_be_bytes());
484        // Terminator: address=0, size=0
485        test_data[16..24].copy_from_slice(&0u64.to_be_bytes());
486        test_data[24..32].copy_from_slice(&0u64.to_be_bytes());
487
488        let iter = MemoryReservationIter {
489            data: &test_data,
490            offset: 0,
491        };
492
493        let reservations: Vec<MemoryReservation, 4> = iter.collect();
494        assert_eq!(reservations.len(), 1);
495        assert_eq!(reservations[0].address, 0x80000000);
496        assert_eq!(reservations[0].size, 0x10000000);
497    }
