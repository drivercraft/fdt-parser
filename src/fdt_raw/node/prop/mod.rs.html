23/// A generic device tree property containing name and raw data.
24///
25/// Represents a property with a name and associated data. Provides methods
26/// for accessing and interpreting the data in various formats (u32, u64,
27/// strings, etc.).
28#[derive(Clone)]
29pub struct Property<'a> {
30    name: &'a str,
31    data: Bytes<'a>,
32}
33
34impl<'a> Property<'a> {
35    /// Creates a new property from a name and data bytes.
36    pub fn new(name: &'a str, data: Bytes<'a>) -> Self {
37        Self { name, data }
38    }
39
40    /// Returns the property name.
41    pub fn name(&self) -> &'a str {
42        self.name
43    }
44
45    /// Returns the property data.
46    pub fn data(&self) -> Bytes<'a> {
47        self.data.clone()
48    }
49
50    /// Returns true if the property has no data.
51    pub fn is_empty(&self) -> bool {
52        self.data.is_empty()
53    }
54
55    /// Returns the length of the property data in bytes.
56    pub fn len(&self) -> usize {
57        self.data.len()
58    }
59
60    /// Returns an iterator over u32 values in the property data.
61    pub fn as_u32_iter(&self) -> U32Iter<'a> {
62        self.data.as_u32_iter()
63    }
64
65    /// Returns an iterator over null-terminated strings in the property data.
66    ///
67    /// Used for properties like `compatible` that contain multiple strings.
68    pub fn as_str_iter(&self) -> StrIter<'a> {
69        self.data.as_str_iter()
70    }
71
72    /// Returns the property data as a byte slice.
73    pub fn as_slice(&self) -> &[u8] {
74        self.data.as_slice()
75    }
76
77    /// Returns the data as a single u64 value.
78    ///
79    /// Returns None if the data is not exactly 8 bytes.
80    pub fn as_u64(&self) -> Option<u64> {
81        let mut iter = self.as_u32_iter();
82        let high = iter.next()? as u64;
83        let low = iter.next()? as u64;
84        if iter.next().is_some() {
85            return None;
86        }
87        Some((high << 32) | low)
88    }
89
90    /// Returns the data as a single u32 value.
91    ///
92    /// Returns None if the data is not exactly 4 bytes.
93    pub fn as_u32(&self) -> Option<u32> {
94        let mut iter = self.as_u32_iter();
95        let value = iter.next()?;
96        if iter.next().is_some() {
97            return None;
98        }
99        Some(value)
100    }
101
102    /// Returns the data as a null-terminated string.
103    pub fn as_str(&self) -> Option<&'a str> {
104        let bytes = self.data.as_slice();
105        let cstr = CStr::from_bytes_until_nul(bytes).ok()?;
106        cstr.to_str().ok()
107    }
108
109    /// Returns the property value as #address-cells.
110    ///
111    /// Only returns a value if the property name is "#address-cells".
112    pub fn as_address_cells(&self) -> Option<u8> {
113        if self.name == "#address-cells" {
114            self.as_u32().map(|v| v as u8)
115        } else {
116            None
117        }
118    }
119
120    /// Returns the property value as #size-cells.
121    ///
122    /// Only returns a value if the property name is "#size-cells".
123    pub fn as_size_cells(&self) -> Option<u8> {
124        if self.name == "#size-cells" {
125            self.as_u32().map(|v| v as u8)
126        } else {
127            None
128        }
129    }
130
131    /// Returns the property value as #interrupt-cells.
132    ///
133    /// Only returns a value if the property name is "#interrupt-cells".
134    pub fn as_interrupt_cells(&self) -> Option<u8> {
135        if self.name == "#interrupt-cells" {
136            self.as_u32().map(|v| v as u8)
137        } else {
138            None
139        }
140    }
141
142    /// Returns the property value as a Status enum.
143    ///
144    /// Only returns a value if the property name is "status".
145    pub fn as_status(&self) -> Option<Status> {
146        let v = self.as_str()?;
147        if self.name == "status" {
148            match v {
149                "okay" | "ok" => Some(Status::Okay),
150                "disabled" => Some(Status::Disabled),
151                _ => None,
152            }
153        } else {
154            None
155        }
156    }
157
158    /// Returns the property value as a phandle.
159    ///
160    /// Only returns a value if the property name is "phandle".
161    pub fn as_phandle(&self) -> Option<Phandle> {
162        if self.name == "phandle" {
163            self.as_u32().map(Phandle::from)
164        } else {
165            None
166        }
167    }
168
169    /// Returns the property value as device_type string.
170    ///
171    /// Only returns a value if the property name is "device_type".
172    pub fn as_device_type(&self) -> Option<&'a str> {
173        if self.name == "device_type" {
174            self.as_str()
175        } else {
176            None
177        }
178    }
179
180    /// Returns the property value as interrupt-parent phandle.
181    ///
182    /// Only returns a value if the property name is "interrupt-parent".
183    pub fn as_interrupt_parent(&self) -> Option<Phandle> {
184        if self.name == "interrupt-parent" {
185            self.as_u32().map(Phandle::from)
186        } else {
187            None
188        }
189    }
190
191    /// Returns the property value as clock-names string list.
192    ///
193    /// Only returns a value if the property name is "clock-names".
194    pub fn as_clock_names(&self) -> Option<StrIter<'a>> {
195        if self.name == "clock-names" {
196            Some(self.as_str_iter())
197        } else {
198            None
199        }
200    }
201
202    /// Returns the property value as compatible string list.
203    ///
204    /// Only returns a value if the property name is "compatible".
205    pub fn as_compatible(&self) -> Option<StrIter<'a>> {
206        if self.name == "compatible" {
207            Some(self.as_str_iter())
208        } else {
209            None
210        }
211    }
212
213    /// Returns true if this is a dma-coherent property.
214    ///
215    /// A dma-coherent property has no data and indicates DMA coherence.
216    pub fn is_dma_coherent(&self) -> bool {
217        self.name == "dma-coherent" && self.data.is_empty()
218    }
219}

221impl fmt::Display for Property<'_> {
222    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
223        if self.is_empty() {
224            write!(f, "{}", self.name())
225        } else if let Some(v) = self.as_address_cells() {
226            write!(f, "#address-cells = <{:#x}>", v)
227        } else if let Some(v) = self.as_size_cells() {
228            write!(f, "#size-cells = <{:#x}>", v)
229        } else if let Some(v) = self.as_interrupt_cells() {
230            write!(f, "#interrupt-cells = <{:#x}>", v)
231        } else if self.name() == "reg" {
232            // reg property needs special handling, but we lack context info
233            // Display raw data
234            write!(f, "reg = ")?;
235            format_bytes(f, &self.data())
236        } else if let Some(s) = self.as_status() {
237            write!(f, "status = \"{:?}\"", s)
238        } else if let Some(p) = self.as_phandle() {
239            write!(f, "phandle = {}", p)
240        } else if let Some(p) = self.as_interrupt_parent() {
241            write!(f, "interrupt-parent = {}", p)
242        } else if let Some(s) = self.as_device_type() {
243            write!(f, "device_type = \"{}\"", s)
244        } else if let Some(iter) = self.as_compatible() {
245            write!(f, "compatible = ")?;
246            let mut first = true;
247            for s in iter.clone() {
248                if !first {
249                    write!(f, ", ")?;
250                }
251                write!(f, "\"{}\"", s)?;
252                first = false;
253            }
254            Ok(())
255        } else if let Some(iter) = self.as_clock_names() {
256            write!(f, "clock-names = ")?;
257            let mut first = true;
258            for s in iter.clone() {
259                if !first {
260                    write!(f, ", ")?;
261                }
262                write!(f, "\"{}\"", s)?;
263                first = false;
264            }
265            Ok(())
266        } else if self.is_dma_coherent() {
267            write!(f, "dma-coherent")
268        } else if let Some(s) = self.as_str() {
269            // Check if there are multiple strings
270            if self.data().iter().filter(|&&b| b == 0).count() > 1 {
271                write!(f, "{} = ", self.name())?;
272                let mut first = true;
273                for s in self.as_str_iter() {
274                    if !first {
275                        write!(f, ", ")?;
276                    }
277                    write!(f, "\"{}\"", s)?;
278                    first = false;
279                }
280                Ok(())
281            } else {
282                write!(f, "{} = \"{}\"", self.name(), s)
283            }
284        } else if self.len() == 4 {
285            // Single u32
286            let v = u32::from_be_bytes(self.data().as_slice().try_into().unwrap());
287            write!(f, "{} = <{:#x}>", self.name(), v)
288        } else {
289            // Raw bytes
290            write!(f, "{} = ", self.name())?;
291            format_bytes(f, &self.data())
292        }
293    }
294}

296/// Formats a byte array as DTS format.
297fn format_bytes(f: &mut fmt::Formatter<'_>, data: &[u8]) -> fmt::Result {
298    if data.len().is_multiple_of(4) {
299        // Format as u32 values
300        write!(f, "<")?;
301        let mut first = true;
302        for chunk in data.chunks(4) {
303            if !first {
304                write!(f, " ")?;
305            }
306            let v = u32::from_be_bytes(chunk.try_into().unwrap());
307            write!(f, "{:#x}", v)?;
308            first = false;
309        }
310        write!(f, ">")
311    } else {
312        // Format as bytes
313        write!(f, "[")?;
314        for (i, b) in data.iter().enumerate() {
315            if i > 0 {
316                write!(f, " ")?;
317            }
318            write!(f, "{:02x}", b)?;
319        }
320        write!(f, "]")
321    }
322}

324/// Property iterator.
325///
326/// Iterates over properties within a node, parsing each property from the
327/// device tree structure block.
328pub struct PropIter<'a> {
329    reader: Reader<'a>,
330    strings: Bytes<'a>,
331    finished: bool,
332}
333
334impl<'a> PropIter<'a> {
335    /// Creates a new property iterator.
336    pub(crate) fn new(reader: Reader<'a>, strings: Bytes<'a>) -> Self {
337        Self {
338            reader,
339            strings,
340
341            finished: false,
342        }
343    }
344
345    /// Handles errors: logs error and terminates iteration.
346    fn handle_error(&mut self, err: FdtError) {
347        error!("Property parse error: {}", err);
348        self.finished = true;
349    }
350
351    /// Reads a property name from the strings block.
352    fn read_prop_name(&self, nameoff: u32) -> Result<&'a str, FdtError> {
353        if nameoff as usize >= self.strings.len() {
354            return Err(FdtError::BufferTooSmall {
355                pos: nameoff as usize,
356            });
357        }
358        let bytes = self.strings.slice(nameoff as usize..self.strings.len());
359        let cstr = CStr::from_bytes_until_nul(bytes.as_slice())?;
360        Ok(cstr.to_str()?)
361    }
362
363    /// Aligns the reader to a 4-byte boundary.
364    fn align4(&mut self) {
365        let pos = self.reader.position();
366        let aligned = (pos + 3) & !3;
367        let skip = aligned - pos;
368        if skip > 0 {
369            let _ = self.reader.read_bytes(skip);
370        }
371    }
372}
373
374impl<'a> Iterator for PropIter<'a> {
375    type Item = Property<'a>;
376
377    fn next(&mut self) -> Option<Self::Item> {
378        if self.finished {
379            return None;
380        }
381
382        loop {
383            let token = match self.reader.read_token() {
384                Ok(t) => t,
385                Err(e) => {
386                    self.handle_error(e);
387                    return None;
388                }
389            };
390
391            match token {
392                Token::Prop => {
393                    // Read property length
394                    let len = match self.reader.read_u32() {
395                        Some(b) => b,
396                        None => {
397                            self.handle_error(FdtError::BufferTooSmall {
398                                pos: self.reader.position(),
399                            });
400                            return None;
401                        }
402                    };
403
404                    // Read property name offset
405                    let nameoff = match self.reader.read_u32() {
406                        Some(b) => b,
407                        None => {
408                            self.handle_error(FdtError::BufferTooSmall {
409                                pos: self.reader.position(),
410                            });
411                            return None;
412                        }
413                    };
414
415                    // Read property data
416                    let prop_data = if len > 0 {
417                        match self.reader.read_bytes(len as _) {
418                            Some(b) => b,
419                            None => {
420                                self.handle_error(FdtError::BufferTooSmall {
421                                    pos: self.reader.position(),
422                                });
423                                return None;
424                            }
425                        }
426                    } else {
427                        Bytes::new(&[])
428                    };
429
430                    // Read property name
431                    let name = match self.read_prop_name(nameoff) {
432                        Ok(n) => n,
433                        Err(e) => {
434                            self.handle_error(e);
435                            return None;
436                        }
437                    };
438
439                    // Align to 4-byte boundary
440                    self.align4();
441
442                    return Some(Property::new(name, prop_data));
443                }
444                Token::BeginNode | Token::EndNode | Token::End => {
445                    // Encountered node boundary, backtrack and terminate property iteration
446                    self.reader.backtrack(4);
447                    self.finished = true;
448                    return None;
449                }
450                Token::Nop => {
451                    // Ignore NOP and continue
452                    continue;
453                }
454                Token::Data(_) => {
455                    // Invalid token
456                    self.handle_error(FdtError::BufferTooSmall {
457                        pos: self.reader.position(),
458                    });
459                    return None;
460                }
461            }
462        }
463    }
