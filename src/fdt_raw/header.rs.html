1//! FDT header parsing.
2//!
3//! This module handles parsing of the Flattened Device Tree header structure,
4//! which appears at the beginning of every device tree blob and contains
5//! metadata about the layout and version of the FDT.
6
7use core::ptr::NonNull;
8
9use crate::FdtError;
10
11/// A 4-byte aligned buffer for header data.
12///
13/// The Device Tree Blob specification requires 4-byte alignment, and this
14/// wrapper ensures that we have properly aligned memory when reading from
15/// potentially unaligned pointers.
16#[repr(align(4))]
17struct AlignedHeader([u8; size_of::<Header>()]);
18
19/// The FDT header structure.
20///
21/// Every device tree blob begins with this header, which contains metadata
22/// about the layout and version of the FDT. All fields are stored in big-endian
23/// byte order on-disk and are converted to host byte order when parsed.
24#[derive(Debug, Clone)]
25pub struct Header {
26    /// FDT header magic number (must be 0xd00dfeed)
27    pub magic: u32,
28    /// Total size in bytes of the FDT structure
29    pub totalsize: u32,
30    /// Offset in bytes from the start of the header to the structure block
31    pub off_dt_struct: u32,
32    /// Offset in bytes from the start of the header to the strings block
33    pub off_dt_strings: u32,
34    /// Offset in bytes from the start of the header to the memory reservation block
35    pub off_mem_rsvmap: u32,
36    /// FDT version number
37    pub version: u32,
38    /// Last compatible FDT version
39    pub last_comp_version: u32,
40    /// Physical ID of the boot CPU
41    pub boot_cpuid_phys: u32,
42    /// Length in bytes of the strings block
43    pub size_dt_strings: u32,
44    /// Length in bytes of the structure block
45    pub size_dt_struct: u32,
46}
47
48impl Header {
49    /// Read a header from a byte slice.
50    ///
51    /// Parses an FDT header from the beginning of a byte slice, validating
52    /// the magic number and converting all fields from big-endian to host order.
53    ///
54    /// # Errors
55    ///
56    /// Returns `FdtError::BufferTooSmall` if the slice is too small to contain
57    /// a complete header, or `FdtError::InvalidMagic` if the magic number doesn't
58    /// match the expected value.
59    pub fn from_bytes(data: &[u8]) -> Result<Self, FdtError> {
60        if data.len() < core::mem::size_of::<Header>() {
61            return Err(FdtError::BufferTooSmall {
62                pos: core::mem::size_of::<Header>(),
63            });
64        }
65        let ptr = NonNull::new(data.as_ptr() as *mut u8).ok_or(FdtError::InvalidPtr)?;
66        unsafe { Self::from_ptr(ptr.as_ptr()) }
67    }
68
69    /// Read a header from a raw pointer.
70    ///
71    /// Parses an FDT header from the memory location pointed to by `ptr`,
72    /// validating the magic number and converting all fields from big-endian
73    /// to host order. Handles unaligned pointers by copying to an aligned buffer.
74    ///
75    /// # Safety
76    ///
77    /// The caller must ensure that the pointer is valid and points to a
78    /// memory region of at least `size_of::<Header>()` bytes that contains a
79    /// valid device tree blob header.
80    ///
81    /// # Errors
82    ///
83    /// Returns `FdtError::InvalidPtr` if the pointer is null, or
84    /// `FdtError::InvalidMagic` if the magic number doesn't match.
85    pub unsafe fn from_ptr(ptr: *mut u8) -> Result<Self, FdtError> {
86        if !(ptr as usize).is_multiple_of(core::mem::align_of::<Header>()) {
87            // Pointer is not aligned, so we need to copy the data to an aligned
88            // buffer first.
89            let mut aligned = AlignedHeader([0u8; core::mem::size_of::<Header>()]);
90            unsafe {
91                core::ptr::copy_nonoverlapping(
92                    ptr,
93                    aligned.0.as_mut_ptr(),
94                    core::mem::size_of::<Header>(),
95                );
96            }
97            Self::from_aligned_ptr(aligned.0.as_mut_ptr())
98        } else {
99            // Pointer is aligned, we can read directly from it.
100            Self::from_aligned_ptr(ptr)
101        }
102    }
103
104    /// Read a header from an aligned pointer.
105    ///
106    /// Internal helper that assumes the pointer is already 4-byte aligned.
107    /// Reads the raw header bytes and converts each field from big-endian.
108    ///
109    /// # Safety
110    ///
111    /// Caller must ensure the pointer is valid, aligned, and points to
112    /// sufficient memory containing a valid FDT header.
113    fn from_aligned_ptr(ptr: *mut u8) -> Result<Self, FdtError> {
114        let ptr = NonNull::new(ptr).ok_or(FdtError::InvalidPtr)?;
115
116        // SAFETY: caller provided a valid pointer to the beginning of a device
117        // tree blob. We read the raw header as it exists in memory (which is
118        // big-endian on-disk). Then convert each u32 field from big-endian to
119        // host order using `u32::from_be`.
120        let raw = unsafe { &*(ptr.cast::<Header>().as_ptr()) };
121
122        let magic = u32::from_be(raw.magic);
123        if magic != crate::FDT_MAGIC {
124            return Err(FdtError::InvalidMagic(magic));
125        }
126
127        Ok(Header {
128            magic,
129            totalsize: u32::from_be(raw.totalsize),
130            off_dt_struct: u32::from_be(raw.off_dt_struct),
131            off_dt_strings: u32::from_be(raw.off_dt_strings),
132            off_mem_rsvmap: u32::from_be(raw.off_mem_rsvmap),
133            version: u32::from_be(raw.version),
134            last_comp_version: u32::from_be(raw.last_comp_version),
135            boot_cpuid_phys: u32::from_be(raw.boot_cpuid_phys),
136            size_dt_strings: u32::from_be(raw.size_dt_strings),
137            size_dt_struct: u32::from_be(raw.size_dt_struct),
138        })
139    }
140}
