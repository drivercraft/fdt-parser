1//! Iterator for traversing all nodes in a Flattened Device Tree.
2//!
3//! This module provides `FdtIter`, which walks through the structure block
4//! of an FDT and yields each node in a depth-first traversal order.
5
6use log::error;
7
8use crate::{
9    Fdt, FdtError, Node, NodeContext, Token,
10    data::{Bytes, Reader},
11    node::{OneNodeIter, OneNodeState},
12};
13
14/// Iterator over all nodes in a Flattened Device Tree.
15///
16/// This iterator performs a depth-first traversal of the device tree,
17/// yielding each node as it's encountered. It maintains a context stack
18/// to track the `#address-cells` and `#size-cells` values inherited from
19/// parent nodes.
20pub struct FdtIter<'a> {
21    fdt: Fdt<'a>,
22    reader: Reader<'a>,
23    strings: Bytes<'a>,
24    /// The node iterator currently being processed
25    node_iter: Option<OneNodeIter<'a>>,
26    /// Whether iteration has terminated (due to error or end)
27    finished: bool,
28    /// Current depth level in the tree
29    level: usize,
30    /// Context stack, with the top being the current context
31    context_stack: heapless::Vec<NodeContext, 16>,
32}
33
34impl<'a> FdtIter<'a> {
35    /// Creates a new FDT iterator from an FDT instance.
36    ///
37    /// Initializes the reader at the start of the structure block and the
38    /// strings slice at the strings block. Also initializes the context
39    /// stack with default values.
40    pub fn new(fdt: Fdt<'a>) -> Self {
41        let header = fdt.header();
42        let struct_offset = header.off_dt_struct as usize;
43        let strings_offset = header.off_dt_strings as usize;
44        let strings_size = header.size_dt_strings as usize;
45
46        let reader = fdt.data.reader_at(struct_offset);
47        let strings = fdt
48            .data
49            .slice(strings_offset..strings_offset + strings_size);
50
51        // Initialize context stack with default context
52        let mut context_stack = heapless::Vec::new();
53        let _ = context_stack.push(NodeContext::default());
54
55        Self {
56            fdt,
57            reader,
58            strings,
59            node_iter: None,
60            level: 0,
61            finished: false,
62            context_stack,
63        }
64    }
65
66    /// Returns the current context (top of the stack).
67    #[inline]
68    fn current_context(&self) -> &NodeContext {
69        // The stack is never empty because we push a default context on initialization
70        self.context_stack.last().unwrap()
71    }
72
73    /// Handles an error by logging it and terminating iteration.
74    fn handle_error(&mut self, err: FdtError) {
75        error!("FDT parse error: {}", err);
76        self.finished = true;
77    }
78}
79
80impl<'a> Iterator for FdtIter<'a> {
81    type Item = Node<'a>;
82
83    fn next(&mut self) -> Option<Self::Item> {
84        if self.finished {
85            return None;
86        }
87
88        loop {
89            // If there's a node being processed, continue processing it
90            if let Some(ref mut node_iter) = self.node_iter {
91                match node_iter.process() {
92                    Ok(OneNodeState::ChildBegin) => {
93                        // Child node encountered, update reader position and clear current node iterator
94                        self.reader = node_iter.reader().clone();
95                        self.node_iter = None;
96                        // Continue loop, next iteration will read BeginNode token
97                    }
98                    Ok(OneNodeState::End) => {
99                        // Current node ended, update reader and decrease level
100                        self.reader = node_iter.reader().clone();
101                        self.node_iter = None;
102                        if self.level > 0 {
103                            self.level -= 1;
104                            // Pop stack to restore parent node context
105                            self.context_stack.pop();
106                        }
107                        // Continue loop to process next token
108                    }
109                    Ok(OneNodeState::Processing) => {
110                        // Should not reach here
111                        continue;
112                    }
113                    Err(e) => {
114                        self.handle_error(e);
115                        return None;
116                    }
117                }
118                continue;
119            }
120
121            // Read next token
122            match self.reader.read_token() {
123                Ok(Token::BeginNode) => {
124                    // Create new node iterator to handle this node
125                    let mut node_iter = OneNodeIter::new(
126                        self.reader.clone(),
127                        self.strings.clone(),
128                        self.level,
129                        self.current_context().clone(),
130                        self.fdt.clone(),
131                    );
132
133                    // Read node name
134                    match node_iter.read_node_name() {
135                        Ok(mut node) => {
136                            // Process node properties to get address-cells, size-cells
137                            match node_iter.process() {
138                                Ok(state) => {
139                                    let props = node_iter.parsed_props();
140
141                                    // Update node's cells
142                                    node.address_cells = props.address_cells.unwrap_or(2);
143                                    node.size_cells = props.size_cells.unwrap_or(1);
144
145                                    // Decide next action based on state
146                                    match state {
147                                        OneNodeState::ChildBegin => {
148                                            // Has child nodes, push child context
149                                            let child_context = NodeContext {
150                                                address_cells: node.address_cells,
151                                                size_cells: node.size_cells,
152                                            };
153                                            let _ = self.context_stack.push(child_context);
154
155                                            // Has child nodes, update reader position
156                                            self.reader = node_iter.reader().clone();
157                                            // Increase level (node has children)
158                                            self.level += 1;
159                                        }
160                                        OneNodeState::End => {
161                                            // Node ended (no children), update reader
162                                            self.reader = node_iter.reader().clone();
163                                            // Don't push or update context since node has no children
164                                            // Don't increase level since node is already closed
165                                        }
166                                        OneNodeState::Processing => {
167                                            // Should not reach here, process() should always return ChildBegin or End
168                                            self.node_iter = Some(node_iter);
169                                            self.level += 1;
170                                        }
171                                    }
172
173                                    return Some(node.into());
174                                }
175                                Err(e) => {
176                                    self.handle_error(e);
177                                    return None;
178                                }
179                            }
180                        }
181                        Err(e) => {
182                            self.handle_error(e);
183                            return None;
184                        }
185                    }
186                }
187                Ok(Token::EndNode) => {
188                    // Top-level EndNode, decrease level
189                    if self.level > 0 {
190                        self.level -= 1;
191                        // Pop stack to restore parent node context
192                        self.context_stack.pop();
193                    }
194                    continue;
195                }
196                Ok(Token::End) => {
197                    // Structure block ended
198                    self.finished = true;
199                    return None;
200                }
201                Ok(Token::Nop) => {
202                    // Ignore NOP tokens
203                    continue;
204                }
205                Ok(Token::Prop) | Ok(Token::Data(_)) => {
206                    // Property or unknown data at top level is an error
207                    self.handle_error(FdtError::BufferTooSmall {
208                        pos: self.reader.position(),
209                    });
210                    return None;
211                }
212                Err(e) => {
213                    self.handle_error(e);
214                    return None;
215                }
216            }
217        }
218    }
219}
