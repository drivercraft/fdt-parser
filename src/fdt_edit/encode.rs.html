1//! FDT encoding module.
2//!
3//! This module handles serialization of the `Fdt` structure into the
4//! DTB (Device Tree Blob) binary format.
5
6use alloc::{string::String, vec::Vec};
7use core::ops::Deref;
8use fdt_raw::{FDT_MAGIC, Token};
9
10use crate::{Fdt, Node};
11
12/// FDT binary data container.
13///
14/// Wraps the encoded DTB data and provides access to the underlying bytes.
15#[derive(Clone, Debug)]
16pub struct FdtData(Vec<u32>);
17
18impl FdtData {
19    /// Returns the data length in bytes.
20    pub fn len(&self) -> usize {
21        self.0.len() * 4
22    }
23
24    /// Returns true if the data is empty.
25    pub fn is_empty(&self) -> bool {
26        self.0.is_empty()
27    }
28}
29
30impl Deref for FdtData {
31    type Target = [u8];
32
33    fn deref(&self) -> &Self::Target {
34        unsafe {
35            core::slice::from_raw_parts(
36                self.0.as_ptr() as *const u8,
37                self.0.len() * core::mem::size_of::<u32>(),
38            )
39        }
40    }
41}
42
43impl AsRef<[u8]> for FdtData {
44    fn as_ref(&self) -> &[u8] {
45        self
46    }
47}

49/// FDT encoder for serializing to DTB format.
50///
51/// This encoder walks the node tree and generates the binary DTB format
52/// according to the Device Tree Specification.
53pub struct FdtEncoder<'a> {
54    fdt: &'a Fdt,
55    struct_data: Vec<u32>,
56    strings_data: Vec<u8>,
57    string_offsets: Vec<(String, u32)>,
58}
59
60impl<'a> FdtEncoder<'a> {
61    /// Creates a new encoder for the given FDT.
62    pub fn new(fdt: &'a Fdt) -> Self {
63        Self {
64            fdt,
65            struct_data: Vec::new(),
66            strings_data: Vec::new(),
67            string_offsets: Vec::new(),
68        }
69    }
70
71    /// Gets or adds a string to the strings block, returning its offset.
72    fn get_or_add_string(&mut self, s: &str) -> u32 {
73        for (existing, offset) in &self.string_offsets {
74            if existing == s {
75                return *offset;
76            }
77        }
78
79        let offset = self.strings_data.len() as u32;
80        self.strings_data.extend_from_slice(s.as_bytes());
81        self.strings_data.push(0); // null terminator
82        self.string_offsets.push((s.into(), offset));
83        offset
84    }
85
86    /// Writes a BEGIN_NODE token and node name.
87    fn write_begin_node(&mut self, name: &str) {
88        let begin_token: u32 = Token::BeginNode.into();
89        self.struct_data.push(begin_token.to_be());
90
91        let name_bytes = name.as_bytes();
92        let name_len = name_bytes.len() + 1; // +1 for null
93        let aligned_len = (name_len + 3) & !3;
94
95        let mut name_buf = vec![0u8; aligned_len];
96        name_buf[..name_bytes.len()].copy_from_slice(name_bytes);
97
98        for chunk in name_buf.chunks(4) {
99            let word = u32::from_ne_bytes(chunk.try_into().unwrap());
100            self.struct_data.push(word);
101        }
102    }
103
104    /// Writes an END_NODE token.
105    fn write_end_node(&mut self) {
106        let end_token: u32 = Token::EndNode.into();
107        self.struct_data.push(end_token.to_be());
108    }
109
110    /// Writes a property to the structure block.
111    fn write_property(&mut self, name: &str, data: &[u8]) {
112        let prop_token: u32 = Token::Prop.into();
113        self.struct_data.push(prop_token.to_be());
114
115        self.struct_data.push((data.len() as u32).to_be());
116
117        let nameoff = self.get_or_add_string(name);
118        self.struct_data.push(nameoff.to_be());
119
120        if !data.is_empty() {
121            let aligned_len = (data.len() + 3) & !3;
122            let mut data_buf = vec![0u8; aligned_len];
123            data_buf[..data.len()].copy_from_slice(data);
124
125            for chunk in data_buf.chunks(4) {
126                let word = u32::from_ne_bytes(chunk.try_into().unwrap());
127                self.struct_data.push(word);
128            }
129        }
130    }
131
132    /// Performs the encoding and returns the binary DTB data.
133    pub fn encode(mut self) -> FdtData {
134        // Recursively encode node tree
135        self.encode_node(&self.fdt.root.clone());
136
137        // Add END token
138        let token: u32 = Token::End.into();
139        self.struct_data.push(token.to_be());
140
141        self.finalize()
142    }
143
144    /// Recursively encodes a node and its children.
145    fn encode_node(&mut self, node: &Node) {
146        // Write BEGIN_NODE and node name
147        self.write_begin_node(node.name());
148
149        // Write all properties (using raw data directly)
150        for prop in node.properties() {
151            self.write_property(prop.name(), &prop.data);
152        }
153
154        // Recursively encode child nodes
155        for child in node.children() {
156            self.encode_node(child);
157        }
158
159        // Write END_NODE
160        self.write_end_node();
161    }
162
163    /// Generates the final FDT binary data.
164    fn finalize(self) -> FdtData {
165        let memory_reservations = &self.fdt.memory_reservations;
166        let boot_cpuid_phys = self.fdt.boot_cpuid_phys;
167
168        let header_size = 40u32; // 10 * 4 bytes
169        let mem_rsv_size = ((memory_reservations.len() + 1) * 16) as u32;
170        let struct_size = (self.struct_data.len() * 4) as u32;
171        let strings_size = self.strings_data.len() as u32;
172
173        let off_mem_rsvmap = header_size;
174        let off_dt_struct = off_mem_rsvmap + mem_rsv_size;
175        let off_dt_strings = off_dt_struct + struct_size;
176        let totalsize = off_dt_strings + strings_size;
177        let totalsize_aligned = (totalsize + 3) & !3;
178
179        let mut data = Vec::with_capacity(totalsize_aligned as usize / 4);
180
181        // Header
182        data.push(FDT_MAGIC.to_be());
183        data.push(totalsize_aligned.to_be());
184        data.push(off_dt_struct.to_be());
185        data.push(off_dt_strings.to_be());
186        data.push(off_mem_rsvmap.to_be());
187        data.push(17u32.to_be()); // version
188        data.push(16u32.to_be()); // last_comp_version
189        data.push(boot_cpuid_phys.to_be());
190        data.push(strings_size.to_be());
191        data.push(struct_size.to_be());
192
193        // Memory reservation block
194        for rsv in memory_reservations {
195            let addr_hi = (rsv.address >> 32) as u32;
196            let addr_lo = rsv.address as u32;
197            let size_hi = (rsv.size >> 32) as u32;
198            let size_lo = rsv.size as u32;
199            data.push(addr_hi.to_be());
200            data.push(addr_lo.to_be());
201            data.push(size_hi.to_be());
202            data.push(size_lo.to_be());
203        }
204        // Terminator
205        data.push(0);
206        data.push(0);
207        data.push(0);
208        data.push(0);
209
210        // Struct block
211        data.extend_from_slice(&self.struct_data);
212
213        // Strings block
214        let strings_aligned_len = (self.strings_data.len() + 3) & !3;
215        let mut strings_buf = vec![0u8; strings_aligned_len];
216        strings_buf[..self.strings_data.len()].copy_from_slice(&self.strings_data);
217
218        for chunk in strings_buf.chunks(4) {
219            let word = u32::from_ne_bytes(chunk.try_into().unwrap());
220            data.push(word);
221        }
222
223        FdtData(data)
224    }
