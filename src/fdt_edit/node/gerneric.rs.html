98    /// Sets the reg property with automatic address translation.
99    ///
100    /// This method converts CPU physical addresses to bus addresses using the
101    /// parent node's ranges mapping before storing them in the reg property.
102    pub fn set_regs(&mut self, regs: &[RegInfo]) {
103        let address_cells = self.ctx.parent_address_cells() as usize;
104        let size_cells = self.ctx.parent_size_cells() as usize;
105        let ranges = self.ctx.current_ranges();
106
107        let mut data = Vec::new();
108
109        for reg in regs {
110            // Convert CPU address to bus address
111            let mut bus_address = reg.address;
112            if let Some(ref ranges) = ranges {
113                for r in ranges {
114                    // Check if CPU address is within ranges mapping range
115                    if reg.address >= r.parent_bus_address
116                        && reg.address < r.parent_bus_address + r.length
117                    {
118                        // Reverse conversion: cpu_address -> bus_address
119                        bus_address = reg.address - r.parent_bus_address + r.child_bus_address;
120                        break;
121                    }
122                }
123            }
124
125            // Write bus address (big-endian)
126            if address_cells == 1 {
127                data.extend_from_slice(&(bus_address as u32).to_be_bytes());
128            } else if address_cells == 2 {
129                data.extend_from_slice(&((bus_address >> 32) as u32).to_be_bytes());
130                data.extend_from_slice(&((bus_address & 0xFFFF_FFFF) as u32).to_be_bytes());
131            }
132
133            // Write size (big-endian)
134            if size_cells == 1 {
135                let size = reg.size.unwrap_or(0);
136                data.extend_from_slice(&(size as u32).to_be_bytes());
137            } else if size_cells == 2 {
138                let size = reg.size.unwrap_or(0);
139                data.extend_from_slice(&((size >> 32) as u32).to_be_bytes());
140                data.extend_from_slice(&((size & 0xFFFF_FFFF) as u32).to_be_bytes());
141            }
142        }
143
144        let prop = Property::new("reg", data);
145        self.node.set_property(prop);
146    }

148    /// Adds a child node to this node.
149    ///
150    /// This method attaches a child node to the current node, updating the
151    /// context to include the parent-child relationship, and returns a
152    /// mutable reference to the newly added child.
153    pub fn add_child(&mut self, child: Node) -> NodeMut<'a> {
154        let name = child.name().to_string();
155        let mut ctx = self.ctx.clone();
156        unsafe {
157            let node_ptr = self.node as *mut Node;
158            let node = &*node_ptr;
159            ctx.push(node);
160        }
161        self.node.add_child(child);
162        let raw = self.node.get_child_mut(&name).unwrap();
163        unsafe {
164            let node_ptr = raw as *mut Node;
165            let node = &mut *node_ptr;
166            NodeMut::new(node, ctx)
167        }
168    }

183/// Internal helper struct for node operations with context.
184///
185/// This struct provides common operations that are shared between
186/// `NodeRefGen` and `NodeMutGen`, avoiding code duplication.
187struct RefOp<'a> {
188    /// Reference to the parsing context
189    ctx: &'a Context<'a>,
190    /// Reference to the node being operated on
191    node: &'a Node,
192}
193
194impl<'a> RefOp<'a> {
195    /// Constructs the full path of the node.
196    ///
197    /// Combines the current context path with the node name to create
198    /// the full device tree path.
199    fn path(&self) -> String {
200        self.ctx.current_path() + "/" + self.node.name()
201    }
202
203    /// Checks if the node's path exactly matches the given path.
204    fn ref_path_eq(&self, path: &str) -> bool {
205        self.path() == path
206    }
207
208    /// Checks if the node's path matches the given path using fuzzy matching.
209    ///
210    /// Fuzzy matching allows comparing paths without requiring the exact
211    /// address portion (the `@address` suffix) to match unless explicitly
212    /// specified. This is useful for matching nodes by name when the
213    /// specific address is not important.
214    fn ref_path_eq_fuzzy(&self, path: &str) -> bool {
215        let mut want = path.trim_matches('/').split("/");
216        let got_path = self.path();
217        let mut got = got_path.trim_matches('/').split("/");
218        let got_count = got.clone().count();
219        let mut current = 0;
220
221        loop {
222            let w = want.next();
223            let g = got.next();
224            let is_last = current + 1 == got_count;
225
226            match (w, g) {
227                (Some(w), Some(g)) => {
228                    if w != g && !is_last {
229                        return false;
230                    }
231
232                    let name = g.split('@').next().unwrap_or(g);
233                    let addr = g.split('@').nth(1);
234
235                    let want_name = w.split('@').next().unwrap_or(w);
236                    let want_addr = w.split('@').nth(1);
237
238                    let res = match (addr, want_addr) {
239                        (Some(a), Some(wa)) => name == want_name && a == wa,
240                        (Some(_), None) => name == want_name,
241                        (None, Some(_)) => false,
242                        (None, None) => name == want_name,
243                    };
244                    if !res {
245                        return false;
246                    }
247                }
248                (None, _) => break,
249                _ => return false,
250            }
251            current += 1;
252        }
253        true
254    }
255
256    /// Parses the reg property and returns a list of register regions.
257    ///
258    /// This method reads the reg property and performs address translation
259    /// from child bus addresses to CPU physical addresses using the parent's
260    /// ranges mapping.
261    fn regs(&self) -> Option<Vec<RegFixed>> {
262        let prop = self.node.get_property("reg")?;
263        let mut iter = prop.as_reader();
264        let address_cells = self.ctx.parent_address_cells() as usize;
265        let size_cells = self.ctx.parent_size_cells() as usize;
266
267        // Get current ranges from context
268        let ranges = self.ctx.current_ranges();
269        let mut out = vec![];
270        let mut size;
271
272        while let Some(mut address) = iter.read_cells(address_cells) {
273            if size_cells > 0 {
274                size = iter.read_cells(size_cells);
275            } else {
276                size = None;
277            }
278            let child_bus_address = address;
279
280            if let Some(ref ranges) = ranges {
281                for r in ranges {
282                    if child_bus_address >= r.child_bus_address
283                        && child_bus_address < r.child_bus_address + r.length
284                    {
285                        address = child_bus_address - r.child_bus_address + r.parent_bus_address;
286                        break;
287                    }
288                }
289            }
290
291            let reg = RegFixed {
292                address,
293                child_bus_address,
294                size,
295            };
296            out.push(reg);
297        }
298
299        Some(out)
300    }
301}
