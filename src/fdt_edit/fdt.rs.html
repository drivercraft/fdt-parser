22/// An editable Flattened Device Tree (FDT).
23///
24/// This structure represents a mutable device tree that can be created from
25/// scratch, loaded from an existing DTB file, modified, and encoded back to
26/// the binary DTB format. It maintains a phandle cache for efficient node
27/// lookups by phandle value.
28#[derive(Clone)]
29pub struct Fdt {
30    /// Boot CPU ID
31    pub boot_cpuid_phys: u32,
32    /// Memory reservation block entries
33    pub memory_reservations: Vec<MemoryReservation>,
34    /// Root node of the device tree
35    pub root: Node,
36    /// Cache mapping phandles to full node paths
37    phandle_cache: BTreeMap<Phandle, String>,
38}
39
40impl Default for Fdt {
41    fn default() -> Self {
42        Self::new()
43    }
44}

58    pub fn from_bytes(data: &[u8]) -> Result<Self, FdtError> {
59        let raw_fdt = fdt_raw::Fdt::from_bytes(data)?;

63    /// Parses an FDT from a raw pointer.
64    ///
65    /// # Safety
66    ///
67    /// The caller must ensure that the pointer is valid and points to a
68    /// valid FDT data structure.
69    pub unsafe fn from_ptr(ptr: *mut u8) -> Result<Self, FdtError> {
70        let raw_fdt = unsafe { fdt_raw::Fdt::from_ptr(ptr)? };
71        Self::from_raw(&raw_fdt)
72    }

74    /// Converts from a raw FDT parser instance.
75    fn from_raw(raw_fdt: &fdt_raw::Fdt) -> Result<Self, FdtError> {
76        let header = raw_fdt.header();
77
78        let mut fdt = Fdt {
79            boot_cpuid_phys: header.boot_cpuid_phys,
80            memory_reservations: raw_fdt.memory_reservations().collect(),
81            root: Node::new(""),
82            phandle_cache: BTreeMap::new(),
83        };
84
85        // Build node tree using a stack to track parent nodes
86        let mut node_stack: Vec<Node> = Vec::new();
87
88        for raw_node in raw_fdt.all_nodes() {
89            let level = raw_node.level();
90            let node = Node::from(&raw_node);
91
92            // Pop stack until we reach the correct parent level
93            while node_stack.len() > level {
94                let child = node_stack.pop().unwrap();
95                if let Some(parent) = node_stack.last_mut() {
96                    parent.add_child(child);
97                } else {
98                    // This is the root node
99                    fdt.root = child;
100                }
101            }
102
103            node_stack.push(node);
104        }
105
106        // Pop all remaining nodes
107        while let Some(child) = node_stack.pop() {
108            if let Some(parent) = node_stack.last_mut() {
109                parent.add_child(child);
110            } else {
111                // This is the root node
112                fdt.root = child;
113            }
114        }
115
116        // Build phandle cache
117        fdt.rebuild_phandle_cache();
118
119        Ok(fdt)
120    }

122    /// Rebuilds the phandle cache by scanning all nodes.
123    pub fn rebuild_phandle_cache(&mut self) {
124        self.phandle_cache.clear();
125        let root_clone = self.root.clone();
126        self.build_phandle_cache_recursive(&root_clone, "/");
127    }
128
129    /// Recursively builds the phandle cache starting from a node.
130    fn build_phandle_cache_recursive(&mut self, node: &Node, current_path: &str) {
131        // Check if node has a phandle property
132        if let Some(phandle) = node.phandle() {
133            self.phandle_cache.insert(phandle, current_path.to_string());
134        }
135
136        // Recursively process child nodes
137        for child in node.children() {
138            let child_name = child.name();
139            let child_path = if current_path == "/" {
140                format!("/{}", child_name)
141            } else {
142                format!("{}/{}", current_path, child_name)
143            };
144            self.build_phandle_cache_recursive(child, &child_path);
145        }
146    }

148    /// Normalizes a path: resolves aliases or ensures leading '/'.
149    fn normalize_path(&self, path: &str) -> Option<String> {
150        if path.starts_with('/') {
151            Some(path.to_string())
152        } else {
153            // Try to resolve as an alias
154            self.resolve_alias(path).map(|s| s.to_string())
155        }
156    }

158    /// Resolves an alias to its full path.
159    ///
160    /// Looks up the alias in the /aliases node and returns the
161    /// corresponding path string.
162    pub fn resolve_alias(&self, alias: &str) -> Option<&str> {
163        let aliases_node = self.get_by_path("/aliases")?;
164        let prop = aliases_node.find_property(alias)?;
165        prop.as_str()
166    }

168    /// Returns all aliases as (name, path) pairs.
169    pub fn aliases(&self) -> Vec<(String, String)> {
170        let mut result = Vec::new();
171        if let Some(aliases_node) = self.get_by_path("/aliases") {
172            for prop in aliases_node.properties() {
173                let name = prop.name().to_string();
174                let path = prop.as_str().unwrap().to_string();
175                result.push((name, path));
176            }
177        }
178        result
179    }

203    /// Applies a device tree overlay to this FDT.
204    ///
205    /// Supports two overlay formats:
206    /// 1. Fragment format: contains fragment@N nodes with target/target-path and __overlay__
207    /// 2. Simple format: directly contains __overlay__ node
208    ///
209    /// # Example
210    ///
211    /// ```ignore
212    /// // Fragment format
213    /// fragment@0 {
214    ///     target-path = "/soc";
215    ///     __overlay__ {
216    ///         new_node { ... };
217    ///     };
218    /// };
219    /// ```
220    pub fn apply_overlay(&mut self, overlay: &Fdt) -> Result<(), FdtError> {
221        // Iterate through all children of overlay root node
222        for child in overlay.root.children() {
223            if child.name().starts_with("fragment@") || child.name() == "fragment" {
224                // Fragment format
225                self.apply_fragment(child)?;
226            } else if child.name() == "__overlay__" {
227                // Simple format: apply directly to root
228                self.merge_overlay_to_root(child)?;
229            } else if child.name() == "__symbols__"
230                || child.name() == "__fixups__"
231                || child.name() == "__local_fixups__"
232            {
233                // Skip these special nodes
234                continue;
235            }
236        }
237
238        // Rebuild phandle cache
239        self.rebuild_phandle_cache();
240
241        Ok(())
242    }
243
244    /// Applies a single fragment from an overlay.
245    fn apply_fragment(&mut self, fragment: &Node) -> Result<(), FdtError> {
246        // Get target path
247        let target_path = self.resolve_fragment_target(fragment)?;
248
249        // Find __overlay__ child node
250        let overlay_node = fragment
251            .get_child("__overlay__")
252            .ok_or(FdtError::NotFound)?;
253
254        // Find target node and apply overlay
255        let target_path_owned = target_path.to_string();
256
257        // Apply overlay to target node
258        self.apply_overlay_to_target(&target_path_owned, overlay_node)?;
259
260        Ok(())
261    }
262
263    /// Resolves the target path of a fragment.
264    fn resolve_fragment_target(&self, fragment: &Node) -> Result<String, FdtError> {
265        // Prefer target-path (string path)
266        if let Some(prop) = fragment.get_property("target-path") {
267            return Ok(prop.as_str().ok_or(FdtError::Utf8Parse)?.to_string());
268        }
269
270        // Use target (phandle reference)
271        if let Some(prop) = fragment.get_property("target") {
272            let ph = prop.get_u32().ok_or(FdtError::InvalidInput)?;
273            let ph = Phandle::from(ph);
274
275            // Find node by phandle and build path
276            if let Some(node) = self.find_by_phandle(ph) {
277                return Ok(node.path());
278            }
279        }
280
281        Err(FdtError::NotFound)
282    }

284    /// Applies an overlay to a target node.
285    fn apply_overlay_to_target(
286        &mut self,
287        target_path: &str,
288        overlay_node: &Node,
289    ) -> Result<(), FdtError> {
290        // Find target node
291        let mut target = self
292            .get_by_path_mut(target_path)
293            .ok_or(FdtError::NotFound)?;
294
295        // Merge overlay properties and child nodes
296        Self::merge_nodes(target.node, overlay_node);
297
298        Ok(())
299    }
300
301    /// Merges an overlay node to the root node.
302    fn merge_overlay_to_root(&mut self, overlay: &Node) -> Result<(), FdtError> {
303        // Merge properties and child nodes to root
304        for prop in overlay.properties() {
305            self.root.set_property(prop.clone());
306        }
307
308        for child in overlay.children() {
309            let child_name = child.name();
310            if let Some(existing) = self.root.get_child_mut(child_name) {
311                // Merge into existing child node
312                Self::merge_nodes(existing, child);
313            } else {
314                // Add new child node
315                self.root.add_child(child.clone());
316            }
317        }
318
319        Ok(())
320    }
321
322    /// Recursively merges two nodes.
323    fn merge_nodes(target: &mut Node, source: &Node) {
324        // Merge properties (source overrides target)
325        for prop in source.properties() {
326            target.set_property(prop.clone());
327        }
328
329        // Merge child nodes
330        for source_child in source.children() {
331            let child_name = &source_child.name();
332            if let Some(target_child) = target.get_child_mut(child_name) {
333                // Recursive merge
334                Self::merge_nodes(target_child, source_child);
335            } else {
336                // Add new child node
337                target.add_child(source_child.clone());
338            }
339        }
340    }

342    /// Applies an overlay with optional deletion of disabled nodes.
343    ///
344    /// If a node in the overlay has status = "disabled", the corresponding
345    /// target node will be disabled or deleted.
346    pub fn apply_overlay_with_delete(
347        &mut self,
348        overlay: &Fdt,
349        delete_disabled: bool,
350    ) -> Result<(), FdtError> {
351        self.apply_overlay(overlay)?;
352
353        if delete_disabled {
354            // Remove all nodes with status = "disabled"
355            Self::remove_disabled_nodes(&mut self.root);
356            self.rebuild_phandle_cache();
357        }
358
359        Ok(())
360    }
361
362    /// Recursively removes disabled nodes.
363    fn remove_disabled_nodes(node: &mut Node) {
364        // Remove disabled child nodes
365        let mut to_remove = Vec::new();
366        for child in node.children() {
367            if matches!(child.status(), Some(Status::Disabled)) {
368                to_remove.push(child.name().to_string());
369            }
370        }
371
372        for child_name in to_remove {
373            node.remove_child(&child_name);
374        }
375
376        // Recursively process remaining child nodes
377        for child in node.children_mut() {
378            Self::remove_disabled_nodes(child);
379        }
380    }

382    /// Removes a node by exact path.
383    ///
384    /// Supports exact path matching only. Aliases are automatically resolved.
385    ///
386    /// # Arguments
387    ///
388    /// * `path` - Node path (e.g., "soc/gpio@1000", "/soc/gpio@1000", or an alias)
389    ///
390    /// # Returns
391    ///
392    /// * `Ok(Some(node))` - The removed node
393    /// * `Ok(None)` - Path not found
394    /// * `Err(FdtError)` - Invalid path format
395    ///
396    /// # Example
397    ///
398    /// ```rust
399    /// # use fdt_edit::{Fdt, Node};
400    /// let mut fdt = Fdt::new();
401    ///
402    /// // Add node then remove it
403    /// let mut soc = Node::new("soc");
404    /// soc.add_child(Node::new("gpio@1000"));
405    /// fdt.root.add_child(soc);
406    ///
407    /// // Remove node with exact path
408    /// let removed = fdt.remove_node("/soc/gpio@1000")?;
409    /// assert!(removed.is_some());
410    /// # Ok::<(), fdt_raw::FdtError>(())
411    /// ```
412    pub fn remove_node(&mut self, path: &str) -> Result<Option<Node>, FdtError> {
413        let normalized_path = self.normalize_path(path).ok_or(FdtError::InvalidInput)?;
414
415        // Use exact path for removal
416        let result = self.root.remove_by_path(&normalized_path)?;
417
418        // If removal succeeded but result is None, path doesn't exist
419        if result.is_none() {
420            return Err(FdtError::NotFound);
421        }
422
423        Ok(result)
424    }

436    /// Finds nodes by path (supports fuzzy matching).
437    pub fn find_by_path<'a>(&'a self, path: &str) -> impl Iterator<Item = NodeRef<'a>> {
438        let path = self
439            .normalize_path(path)
440            .unwrap_or_else(|| path.to_string());
441
442        NodeIter::new(&self.root).filter_map(move |node_ref| {
443            if node_ref.path_eq_fuzzy(&path) {
444                Some(node_ref)
445            } else {
446                None
447            }
448        })
449    }

475    /// Finds nodes with matching compatible strings.
476    pub fn find_compatible(&self, compatible: &[&str]) -> Vec<NodeRef<'_>> {
477        let mut results = Vec::new();
478        for node_ref in self.all_nodes() {
479            let Some(ls) = node_ref.compatible() else {
480                continue;
481            };
482
483            for comp in ls {
484                if compatible.contains(&comp) {
485                    results.push(node_ref);
486                    break;
487                }
488            }
489        }
490        results
491    }

499impl core::fmt::Display for Fdt {
500    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
501        // Output DTS header
502        writeln!(f, "/dts-v1/;")?;
503
504        // Output memory reservation block
505        for reservation in &self.memory_reservations {
506            writeln!(
507                f,
508                "/memreserve/ 0x{:x} 0x{:x};",
509                reservation.address, reservation.size
510            )?;
511        }
512
513        // Output root node
514        writeln!(f, "{}", self.root)
515    }
516}

518impl core::fmt::Debug for Fdt {
519    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
520        if f.alternate() {
521            // Deep debug format with node traversal
522            self.fmt_debug_deep(f)
523        } else {
524            // Simple debug format
525            f.debug_struct("Fdt")
526                .field("boot_cpuid_phys", &self.boot_cpuid_phys)
527                .field("memory_reservations_count", &self.memory_reservations.len())
528                .field("root_node_name", &self.root.name)
529                .field("total_nodes", &self.root.children().len())
530                .field("phandle_cache_size", &self.phandle_cache.len())
531                .finish()
532        }
533    }

536impl Fdt {
537    /// Formats the FDT with detailed debug information.
538    fn fmt_debug_deep(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
539        writeln!(f, "Fdt {{")?;
540        writeln!(f, "    boot_cpuid_phys: 0x{:x},", self.boot_cpuid_phys)?;
541        writeln!(
542            f,
543            "    memory_reservations_count: {},",
544            self.memory_reservations.len()
545        )?;
546        writeln!(f, "    phandle_cache_size: {},", self.phandle_cache.len())?;
547        writeln!(f, "    nodes:")?;
548
549        // Iterate through all nodes and print debug info with indentation
550        for (i, node) in self.all_nodes().enumerate() {
551            self.fmt_node_debug(f, &node, 2, i)?;
552        }
553
554        writeln!(f, "}}")
555    }
556
557    /// Formats a single node for debug output.
558    fn fmt_node_debug(
559        &self,
560        f: &mut core::fmt::Formatter<'_>,
561        node: &NodeRef,
562        indent: usize,
563        index: usize,
564    ) -> core::fmt::Result {
565        // Print indentation
566        for _ in 0..indent {
567            write!(f, "    ")?;
568        }
569
570        // Print node index and basic info
571        write!(f, "[{:03}] {}: ", index, node.name())?;
572
573        // Print type-specific information
574        match node.as_ref() {
575            NodeKind::Clock(clock) => {
576                write!(f, "Clock")?;
577                if let ClockType::Fixed(fixed) = &clock.kind {
578                    write!(f, " (Fixed, {}Hz)", fixed.frequency)?;
579                } else {
580                    write!(f, " (Provider)")?;
581                }
582                if !clock.clock_output_names.is_empty() {
583                    write!(f, ", outputs: {:?}", clock.clock_output_names)?;
584                }
585                write!(f, ", cells={}", clock.clock_cells)?;
586            }
587            NodeKind::Pci(pci) => {
588                write!(f, "PCI")?;
589                if let Some(bus_range) = pci.bus_range() {
590                    write!(f, " (bus: {:?})", bus_range)?;
591                }
592                write!(f, ", interrupt-cells={}", pci.interrupt_cells())?;
593            }
594            NodeKind::InterruptController(ic) => {
595                write!(f, "InterruptController")?;
596                if let Some(cells) = ic.interrupt_cells() {
597                    write!(f, " (cells={})", cells)?;
598                }
599                let compatibles = ic.compatibles();
600                if !compatibles.is_empty() {
601                    write!(f, ", compatible: {:?}", compatibles)?;
602                }
603            }
604            NodeKind::Memory(mem) => {
605                write!(f, "Memory")?;
606                let regions = mem.regions();
607                if !regions.is_empty() {
608                    write!(f, " ({} regions", regions.len())?;
609                    for (i, region) in regions.iter().take(2).enumerate() {
610                        write!(f, ", [{}]: 0x{:x}+0x{:x}", i, region.address, region.size)?;
611                    }
612                    if regions.len() > 2 {
613                        write!(f, ", ...")?;
614                    }
615                    write!(f, ")")?;
616                }
617            }
618            NodeKind::Generic(_) => {
619                write!(f, "Generic")?;
620            }
621        }
622
623        // Print phandle information
624        if let Some(phandle) = node.phandle() {
625            write!(f, ", phandle={}", phandle)?;
626        }
627
628        // Print address and size cells information
629        if let Some(address_cells) = node.address_cells() {
630            write!(f, ", #address-cells={}", address_cells)?;
631        }
632        if let Some(size_cells) = node.size_cells() {
633            write!(f, ", #size-cells={}", size_cells)?;
634        }
635
636        writeln!(f)?;
637
638        Ok(())
639    }
